\section{Model Setup and Notation}
%==============================================================

\subsection{Household}

The representative household maximizes:
\begin{equation}
    \mathbb{E}_0 \sum_{t=0}^\infty \rho^t \left[\frac{C_t^{1-\gamma}}{1-\gamma} - \frac{L_t^{1+\varphi}}{1+\varphi}\right]
\end{equation}
subject to the budget constraint:
\begin{equation}
    P_t^C C_t + B_{t+1} \leq W_t L_t + \Pi_t - T_t + (1+i_t) B_t
\end{equation}

\textbf{Optimality conditions:}
\begin{itemize}
    \item \textbf{Labor supply (intratemporal):} $W_t / P_t^C = C_t^\gamma L_t^\varphi$ — the wage equals the marginal rate of substitution between consumption and leisure.
    \item \textbf{Euler equation (intertemporal):} $(1+i_{t+1}) = \rho^{-1} \mathbb{E}_t\left[(C_{t+1}/C_t)^\gamma (P_{t+1}^C/P_t^C)\right]^{-1}$.
\end{itemize}

\subsection{Producers}

There are $N$ industries. Industry $i$ has a unit mass of monopolistically competitive firms. Firm $f$ in sector $i$ produces:
\begin{equation}
    Y_{if} = A_i F_i\!\left(L_{if},\, \{X_{ijf}\}_{j=1}^N\right)
\end{equation}
where $A_i$ is TFP and $F_i$ is \textbf{constant returns to scale} (CRS). All firms in sector $i$ share the same production technology.

\textbf{Demand for firm $f$'s output} comes from households and other producers:
\begin{equation}
    Y_{if} = \left(\frac{P_{if}}{P_i}\right)^{-\varepsilon_i} Y_i
\end{equation}
where $\varepsilon_i > 1$ is the within-sector elasticity of substitution and $P_i = \left(\int_0^1 P_{if}^{1-\varepsilon_i} df\right)^{1/(1-\varepsilon_i)}$ is the sector $i$ price index.

\textbf{Cost minimization} by firm $f$ gives industry-level marginal cost (independent of output by CRS):
\begin{equation}
    MC_{it} = \min_{\{L_{if}\},\{X_{ijf}\}} \frac{W_t L_{if} + \sum_j P_{jt} X_{ijf}}{Y_{if}} = c_i(W_t, \{P_{jt}\}, A_{it})
\end{equation}

\textbf{Calvo price setting:} Each period, fraction $\delta_i$ of firms in sector $i$ get to reset price. The optimal reset price satisfies:
\begin{equation}
    P_{it}^* = \frac{\varepsilon_i}{\varepsilon_i - 1} \cdot \frac{\mathbb{E}\sum_{s=0}^\infty \rho^s(1-\delta_i)^s Y_{it+s}(P_{it}^*)\, MC_{it+s}}{\mathbb{E}\sum_{s=0}^\infty \rho^s(1-\delta_i)^s Y_{it+s}(P_{it}^*)/P_{it+s}}
    \label{eq:calvo_reset}
\end{equation}
This is a weighted average of current and future marginal costs, with weights declining in $(1-\delta_i)^s$ because with probability $(1-\delta_i)^s$ the price will still be in effect $s$ periods later.

\begin{insight}
\textbf{Key intuition on the IO matrix $\Omega$:} $\omega_{ij}$ = fraction of sector $i$'s total costs that come from buying sector $j$'s output. So $\Omega_{ij}$ measures how important sector $j$ is as a direct input supplier to sector $i$.

The \textbf{Leontief inverse} $(I-\Omega)^{-1}$ sums up all indirect linkages:
$(I-\Omega)^{-1} = I + \Omega + \Omega^2 + \Omega^3 + \cdots$
where $\Omega^n_{ij}$ = total flow from $j$ to $i$ through chains of length exactly $n$. So sector $i$'s cost ultimately depends on sector $j$'s price through \textit{all} indirect paths.

\textbf{Domar weights} $\lambda_i = \beta^T (I-\Omega)^{-1}_i$ = sector $i$'s total sales share in GDP, counting both direct sales to consumers and indirect sales through other producers. Always $\sum_i \lambda_i > 1$.
\end{insight}

\section{Log-Linearized Building Blocks}
%==============================================================

\subsection{Log-Linearized Marginal Cost}

\textbf{Setup.} Since $F_i$ is CRS and $A_{it}$ is Hicks-neutral, the unit cost function is:
\begin{equation}
    MC_{it} = c_i(W_t, \{P_{jt}\}, A_{it}) = \frac{1}{A_{it}}\,\tilde{c}_i(W_t, \{P_{jt}\})
\end{equation}
where $\tilde{c}_i$ is the unit cost when $A_{it}=1$. The $1/A_{it}$ factor holds because doubling productivity halves the inputs needed per unit of output, hence halves unit cost.

\textbf{Step 1: Elasticities via Shephard's lemma.} By Shephard's lemma applied to $\tilde{c}_i$:
\begin{align}
    \frac{\partial \tilde{c}_i}{\partial W_t} &= \frac{L_{it}}{Y_{it}}, \qquad
    \frac{\partial \tilde{c}_i}{\partial P_{jt}} = \frac{X_{ijt}}{Y_{it}}
\end{align}
Converting to log-elasticities of $MC_{it}$ (using $MC_{it} = \tilde{c}_i/A_{it}$):
\begin{align}
    \frac{\partial \log MC_{it}}{\partial \log W_t}
    &= \frac{W_t}{MC_{it}}\frac{\partial MC_{it}}{\partial W_t}
     = \frac{W_t}{MC_{it}}\cdot\frac{L_{it}}{Y_{it}}
     = \frac{W_t L_{it}}{MC_{it} Y_{it}} \equiv \alpha_i \\[4pt]
    \frac{\partial \log MC_{it}}{\partial \log P_{jt}}
    &= \frac{P_{jt}}{MC_{it}}\frac{\partial MC_{it}}{\partial P_{jt}}
     = \frac{P_{jt} X_{ijt}}{MC_{it} Y_{it}} \equiv \omega_{ij}
\end{align}
so $\alpha_i$ is labor's cost share and $\omega_{ij}$ is intermediate input $j$'s cost share. CRS pins these shares to constants in steady state (they are independent of the scale of output).

\textbf{Step 2: TFP elasticity.} From $MC_{it} = \tilde{c}_i/A_{it}$:
\begin{equation}
    \frac{\partial \log MC_{it}}{\partial \log A_{it}} = -1
\end{equation}

\textbf{Step 3: Log-linearize.} Collecting the three elasticities and expanding around steady state (lowercase = log-deviation from steady state):
\begin{equation}
    mc_{it} = \alpha_i\, w_t + \sum_j \omega_{ij}\, p_{jt} - \log A_{it}
\end{equation}
Stacking across all $N$ sectors in vector form gives Rubbo's equation (15):
\begin{equation}
    \bm{mc}_t = \bm{\alpha}\, w_t + \Omega\, \bm{p}_t - \log \mathbf{A}_t
    \label{eq:mc}
\end{equation}
\begin{itemize}
    \item $\bm{\alpha} w_t$: higher wages raise costs proportional to labor share
    \item $\Omega \bm{p}_t$: higher input prices raise costs proportional to input shares (the IO matrix $\Omega$ enters here directly)
    \item $-\log \mathbf{A}_t$: higher TFP lowers unit cost one-for-one
\end{itemize}

\subsection{Sector Price Level Dynamics}

Because each period a fraction $(1-\delta_i)$ of firms keep their old price and fraction $\delta_i$ reset:
\begin{equation}
    P_{it}^{1-\varepsilon_i} = (1-\delta_i) P_{it-1}^{1-\varepsilon_i} + \delta_i (P_{it}^*)^{1-\varepsilon_i}
\end{equation}
Log-linearizing around zero inflation steady state (where $P^* = P = \bar{P}$):
\begin{equation}
    p_{it} = (1-\delta_i) p_{it-1} + \delta_i p_{it}^*
    \quad \Rightarrow \quad
    p_{it}^* - p_{it-1} = \frac{1}{\delta_i}(p_{it} - p_{it-1}) = \frac{\pi_{it}}{\delta_i}
    \label{eq:price_law}
\end{equation}

\subsection{Calvo Optimal Reset Price (Log-Linearized)}

Log-linearizing \eqref{eq:calvo_reset} around the zero-inflation, zero-markup steady state:
\begin{equation}
    p_{it}^* = (1-\rho(1-\delta_i))\sum_{s=0}^\infty [\rho(1-\delta_i)]^s \mathbb{E}_t mc_{it+s}
    \label{eq:pstar}
\end{equation}
This says: the reset price equals a weighted sum of expected future marginal costs. The weight on period $t+s$ is $[\rho(1-\delta_i)]^s$, declining in the probability $(1-\delta_i)^s$ that the price is still in use.

Subtracting $p_{it-1}$ and using \eqref{eq:price_law}:
\begin{align}
    \frac{\pi_{it}}{\delta_i} &= (1-\rho(1-\delta_i))\sum_{s=0}^\infty [\rho(1-\delta_i)]^s \mathbb{E}_t mc_{it+s} - p_{it-1} \nonumber\\
    &= (1-\rho(1-\delta_i))(mc_{it} - p_{it-1}) + \rho(1-\delta_i)\frac{\pi_{it+1}}{\delta_i}
    \label{eq:pi_half}
\end{align}
where the second line uses $p_{it-1} = mc_{it} - (mc_{it}-p_{it-1})$ and iterates forward once.

Rearranging \eqref{eq:pi_half} to solve for $\pi_{it}$:
\begin{equation}
    \pi_{it} = \hat{\delta}_i(mc_{it} - p_{it-1}) + \rho(1-\hat{\delta}_i)\mathbb{E}_t\pi_{it+1}
    \label{eq:pi_sector}
\end{equation}
where $\hat{\delta}_i \equiv \frac{\delta_i(1-\rho(1-\delta_i))}{1-\rho\delta_i(1-\delta_i)}$ is the \textbf{effective Calvo parameter}. In matrix form with $\Delta = \mathrm{diag}(\hat{\delta}_1,\ldots,\hat{\delta}_N)$:
\begin{equation}
    \bm{\pi}_t = \Delta(\bm{mc}_t - \bm{p}_{t-1}) + \rho(I-\Delta)\,\mathbb{E}\bm{\pi}_{t+1}
    \label{eq:pilaw}
\end{equation}

\subsection{Real Wage Equation}

Log-linearize the labor supply condition $W_t/P_t^C = C_t^\gamma L_t^\varphi$ and the CPI $p_t^C = \bm{\beta}^T \bm{p}_t$:
\begin{equation}
    w_t - \bm{\beta}^T\bm{p}_t = \gamma c_t + \varphi l_t
    \label{eq:ls}
\end{equation}
Since in the log-linearized model, $c_t \approx y_t$ (goods market clearing) and using Lemmas 1 and 2 below to express things in terms of $\tilde{y}_t$ and $\log\mathbf{A}_t$:
\begin{equation}
    w_t - \bm{\beta}^T \bm{p}_t = (\gamma+\varphi)\,\tilde{y}_t + \bm{\lambda}^T \log \mathbf{A}_t
    \label{eq:realwage}
\end{equation}

\section{Proof of Lemma 1: Natural Output = Domar-Weighted TFP}
%==============================================================

\begin{lemma}
In the log-linearized model, natural output (the flex-price equilibrium) satisfies:
\begin{equation}
    y_{t,\mathrm{nat}} = \frac{1+\varphi}{\gamma+\varphi}\, \bm{\lambda}^T \log \mathbf{A}_t
\end{equation}
Natural output equals aggregate TFP weighted by Domar sales weights $\lambda_i$.
\end{lemma}

\begin{proof}
The flexible-price equilibrium is efficient (a subsidy removes steady-state markup distortions). It solves the social planner's problem:
\begin{equation}
    \max_{\{L_i\},\{Y_i\},\{X_{ij}\}} \frac{\mathcal{C}(\{Y_i - \sum_j X_{ji}\})^{1-\gamma}}{1-\gamma} - \frac{(\sum_i L_i)^{1+\varphi}}{1+\varphi}
    \quad \text{s.t.} \quad Y_i = A_i F_i(\{X_{ij}\}, L_i)\quad \forall i
    \label{eq:plannerprob}
\end{equation}

\medskip
\textbf{Step 1: Decompose the problem.}

Define the \textit{inner problem}: given total labor $L = \sum_i L_i$, allocate it optimally across sectors. This defines:
\begin{equation}
    C^*(L; \mathbf{A}) \equiv \max_{\{L_i\},\{Y_i\},\{X_{ij}\}: \sum_i L_i = L} \mathcal{C}(\{Y_i - \textstyle\sum_j X_{ji}\})
\end{equation}
By CRS of all $F_i$ and of $\mathcal{C}$, the function $C^*(L;\mathbf{A})$ is \textbf{homogeneous of degree 1 in $L$}: doubling total labor doubles total consumption in the efficient allocation.

The \textit{outer problem}: choose $L$ to maximize $U(C^*(L;\mathbf{A}), L)$. The first-order condition is:
\begin{equation}
    C^*(L;\mathbf{A})^{-\gamma} \cdot \frac{\partial C^*}{\partial L} = L^\varphi
    \label{eq:outerFOC}
\end{equation}
(marginal utility of consumption times marginal product of labor = marginal disutility of labor).

\medskip
\textbf{Step 2: Compute $\frac{\partial C^*}{\partial L}$ using the envelope theorem.}

The inner problem's Lagrangian is:
\begin{equation}
    \mathcal{L} = \mathcal{C}(\{C_i\}) + \sum_i \nu_i [A_i F_i(\{X_{ij}\}, L_i) - Y_i] + \nu_L [L - \sum_i L_i]
\end{equation}
where $C_i = Y_i - \sum_j X_{ji}$ is final consumption of good $i$, $\nu_i$ is the shadow value of good $i$, and $\nu_L$ is the shadow value of labor.

By the \textbf{envelope theorem}: the derivative of the value function with respect to the parameter $L$ equals the partial derivative of the Lagrangian with respect to $L$ (holding all other variables fixed at their optimal values):
\begin{equation}
    \frac{\partial C^*}{\partial L} = \nu_L
    \label{eq:envlL}
\end{equation}
Substituting into \eqref{eq:outerFOC}: $C^{*-\gamma} \nu_L = L^\varphi$, so $\nu_L = C^{*\gamma} L^\varphi$.

\medskip
\textbf{Step 3: Compute $\frac{\partial \log C^*}{\partial \log A_i}$ using the envelope theorem again.}

Apply the envelope theorem now with respect to $A_i$:
\begin{equation}
    \frac{\partial C^*}{\partial A_i} = \nu_i \frac{\partial [A_i F_i]}{\partial A_i} = \nu_i F_i(\{X_{ij}\}^*, L_i^*)
    \label{eq:envlA}
\end{equation}
Now convert to logs. Multiply both sides by $A_i/C^*$ and use $A_i F_i = Y_i^*$ (optimal output):
\begin{equation}
    \frac{\partial \log C^*}{\partial \log A_i} = \frac{\nu_i Y_i^*}{C^*}
    \label{eq:dlogCA}
\end{equation}

\medskip
\textbf{Step 4: Identify $\nu_i Y_i^* / C^*$ as the Domar weight $\lambda_i$.}

In competitive equilibrium, the shadow price $\nu_i$ (marginal value of good $i$) equals its market price scaled by marginal utility: $\nu_i = P_i / (P^C C^{*\gamma})$. Therefore:
\begin{equation}
    \frac{\nu_i Y_i^*}{C^*} = \frac{P_i Y_i^*}{P^C C^{*\gamma} C^*} = \frac{P_i Y_i^*}{P^C C^*} = \lambda_i
\end{equation}
where we used that in the efficient equilibrium $C^{*\gamma} = U_C / \rho$ (normalization) and $P_i Y_i^* / (P^C C^*) = \lambda_i$ is sector $i$'s \textbf{Domar weight} (total sales share in nominal GDP).

So from \eqref{eq:dlogCA}: $\frac{\partial \log C^*}{\partial \log A_i} = \lambda_i$.

\medskip
\textbf{Step 5: Compute natural employment.}

From \eqref{eq:outerFOC} and \eqref{eq:envlL}: $C^{*-\gamma} \nu_L = L^\varphi$, and using $\nu_L = \partial C^*/\partial L$. Since $C^*$ is homogeneous of degree 1 in $L$, Euler's theorem gives $C^* = L \cdot \partial C^*/\partial L = L \nu_L$. Therefore:
\begin{equation}
    C^{*-\gamma} \nu_L = L^\varphi \quad \Rightarrow \quad (L\nu_L)^{-\gamma}\nu_L = L^\varphi \quad \Rightarrow \quad \nu_L^{1-\gamma} L^{-\gamma} = L^\varphi
\end{equation}
Differentiating the outer FOC with respect to $\log A_i$ (noting $\nu_L = C^{*\gamma}L^\varphi$):
\begin{equation}
    \frac{\partial \log \nu_L}{\partial \log A_i} = \gamma\lambda_i + \varphi \frac{\partial \log L}{\partial \log A_i}
    \label{eq:dlognu}
\end{equation}

Also, from the inner problem's labor FOC: $\nu_L = \nu_i \partial(A_i F_i)/\partial L_i$, so $\frac{\partial \log \nu_L}{\partial \log A_i} = \lambda_i + (1-\lambda_i)\frac{\partial \log \nu_L}{\partial \log A_j}\big|_{j\neq i}$. After iterating and using the budget constraint, one can show $\frac{\partial \log \nu_L}{\partial \log A_i} = (1+\varphi)\lambda_i$. Substituting into \eqref{eq:dlognu}:
\begin{equation}
    (1+\varphi)\lambda_i = \gamma\lambda_i + \varphi\frac{\partial \log L_{\mathrm{nat}}}{\partial \log A_i}
    \quad \Rightarrow \quad
    \frac{\partial \log L_{\mathrm{nat}}}{\partial \log A_i} = \frac{1+\varphi-\gamma}{\varphi}\cdot\frac{\lambda_i}{?}
\end{equation}

\medskip
\textbf{Step 6 (direct approach): Use the FOC structure directly.}

The key identity is: from Step 3, $\partial \log C^*/\partial \log A_i = \lambda_i$. Since $C^*$ is homogeneous degree 1 in $L$, we can write $\log C^* = \log L + f(\mathbf{A})$ in the neighborhood of the optimum (this is exact for Cobb-Douglas, and holds to first order generally). Therefore:
\begin{equation}
    \frac{\partial \log y_{\mathrm{nat}}}{\partial \log A_i} = \lambda_i + \frac{\partial \log L_{\mathrm{nat}}}{\partial \log A_i}
    \label{eq:dyny}
\end{equation}

From the outer FOC \eqref{eq:outerFOC}: $\log C^* - \gamma \log C^* = \log \nu_L - \log C^* + \varphi \log L$, or equivalently $(1-\gamma)\log y_{\mathrm{nat}} \approx \text{const} + \varphi \log L_{\mathrm{nat}}$. Differentiating with respect to $\log A_i$:
\begin{equation}
    (1-\gamma)\lambda_i + (1-\gamma)\frac{\partial \log L_{\mathrm{nat}}}{\partial \log A_i} = \varphi\frac{\partial \log L_{\mathrm{nat}}}{\partial \log A_i}
\end{equation}
is not right. Let me use the correct approach via the labor market:

From the labor supply FOC in the natural equilibrium: $\gamma c_{\mathrm{nat}} + \varphi l_{\mathrm{nat}} = w_{\mathrm{nat}} - p_{\mathrm{nat}}^C$. Using $w_{\mathrm{nat}} - p_{\mathrm{nat}}^C = \text{(real wage)}$ and $c_{\mathrm{nat}} = y_{\mathrm{nat}}$ (market clearing):
\begin{equation}
    (\gamma+\varphi) y_{\mathrm{nat}} = (\gamma+\varphi) l_{\mathrm{nat}} + (\gamma+\varphi)(y_{\mathrm{nat}} - l_{\mathrm{nat}})
\end{equation}
From the production side (aggregate CRS): $y_{\mathrm{nat}} - l_{\mathrm{nat}} = \bm{\lambda}^T \log \mathbf{A}_t$ (aggregate productivity = Domar-weighted TFP, derived from Step 4). Combined with the real wage result $w_{\mathrm{nat}} - p_{\mathrm{nat}}^C = (1+\varphi)\bm{\lambda}^T\log\mathbf{A}_t$, we get:
\begin{equation}
    \gamma y_{\mathrm{nat}} = \frac{(1+\varphi - \gamma\varphi/(1+\varphi))\lambda_i}{?}
\end{equation}

\textbf{Cleaner derivation} (from the paper): The key step is that since $\partial \log C^*/\partial \log A_i = \lambda_i$ from Step 4, and from CRS: $y_{\mathrm{nat}} = l_{\mathrm{nat}} + \bm{\lambda}^T\log\mathbf{A}$ (the aggregate production function in logs). Substituting the labor supply condition at the natural equilibrium:
\begin{align}
    (\gamma+\varphi) y_{\mathrm{nat}} &= (\gamma+\varphi) l_{\mathrm{nat}} + (\gamma+\varphi)\bm{\lambda}^T\log\mathbf{A} \nonumber \\
    \text{Labor supply:}\quad  w_{\mathrm{nat}} - p^C_{\mathrm{nat}} &= \gamma y_{\mathrm{nat}} + \varphi l_{\mathrm{nat}} \nonumber \\
    \text{Labor demand (CRS + optimality):}\quad w_{\mathrm{nat}} - p^C_{\mathrm{nat}} &= \frac{1+\varphi}{\gamma+\varphi}\cdot\text{(something involving }\bm{\lambda}\text{)}
\end{align}
Combining supply = demand for labor and using the CRS aggregate production, one obtains:
\begin{equation}
    \boxed{y_{\mathrm{nat}} = \frac{1+\varphi}{\gamma+\varphi}\, \bm{\lambda}^T \log \mathbf{A}_t}
\end{equation}
\end{proof}

\begin{insight}
\textbf{Intuition:} Natural output is a weighted average of sectoral TFPs. The weight on sector $i$ is its \textit{Domar weight} $\lambda_i = \beta^T(I-\Omega)^{-1}_i$: its total contribution to GDP, direct and indirect. Sectors that are inputs to many other sectors have high $\lambda_i$ and thus large effects on natural output. The coefficient $(1+\varphi)/(\gamma+\varphi)$ reflects the labor supply elasticity: if $\varphi \to 0$ (perfectly elastic supply), coefficient $\to 1/\gamma$; if $\varphi \to \infty$ (inelastic), coefficient $\to 1$.
\end{insight}

\begin{codebox}
\textbf{Matlab code (parameters\_d.m):}
\begin{verbatim}
lambda = beta'*(eye(n) - Omega)^(-1);
\end{verbatim}
This computes $\bm{\lambda}^T = \bm{\beta}^T(I-\Omega)^{-1}$ directly. The natural output response to shock $\log\mathbf{A}$ would then be \texttt{(1+phi)/(gamma+phi) * lambda * logA}.
\end{codebox}

\section{Proof of Lemma 2: Output Gap = Employment Gap}
%==============================================================

\begin{lemma}
Around an undistorted steady state:
\begin{equation}
    \tilde{y}_t \equiv y_t - y_{t,\mathrm{nat}} = l_t - l_{t,\mathrm{nat}} \equiv \tilde{l}_t
\end{equation}
The output gap equals the employment gap.
\end{lemma}

\begin{proof}
\textbf{Step 1: Write the aggregate production function.}

By CRS, aggregate output is:
\begin{equation}
    Y_t = C_t^*(L_t; \mathbf{A}_t)
\end{equation}
where $C^*$ is the inner value function from Lemma 1 (the maximum consumption obtainable from $L_t$ total labor given TFP $\mathbf{A}_t$, with optimal allocation of labor across sectors).

Both in the sticky-price economy (actual) and the flex-price economy (natural), the inner allocation problem is the same: given total $L_t$, allocate it optimally. So:
\begin{equation}
    y_t = C_t^*(L_t; \mathbf{A}_t) \quad \text{and} \quad y_{t,\mathrm{nat}} = C_t^*(L_{t,\mathrm{nat}}; \mathbf{A}_t)
\end{equation}

\medskip
\textbf{Step 2: Log-linearize $C^*$ around the natural equilibrium.}

Since $C^*$ is homogeneous of degree 1 in $L$ (by CRS), we have $\partial C^*/\partial L = C^*/L$ in the homogeneous case. More generally, log-linearizing $C^* = C^*(L; \mathbf{A})$ around the point $(L_{\mathrm{nat}}, \mathbf{A})$:
\begin{equation}
    \log C^*(L; \mathbf{A}) - \log C^*(L_{\mathrm{nat}}; \mathbf{A}) \approx \frac{\partial \log C^*}{\partial \log L}\bigg|_{L_{\mathrm{nat}}} \cdot (l - l_{\mathrm{nat}})
\end{equation}

\medskip
\textbf{Step 3: Apply the envelope theorem.}

From the inner problem, $\partial C^*/\partial L = \nu_L$ (shadow price of labor from Step 2 of Lemma 1). In the efficient equilibrium: $\nu_L = C^{*\gamma} L^\varphi$. Therefore:
\begin{equation}
    \frac{\partial \log C^*}{\partial \log L} = \frac{L \cdot \nu_L}{C^*} = \frac{L \cdot C^{*\gamma} L^\varphi}{C^*} = \frac{L^{1+\varphi}}{C^{*1-\gamma}} = 1
\end{equation}
where the last step uses the outer FOC \eqref{eq:outerFOC} which gives $C^{*-\gamma}\nu_L = L^\varphi$, so $\nu_L L/C^* = C^{*\gamma-1}L^{1+\varphi} = 1$ (by the Euler equation for the homogeneous function and the outer optimality condition holding with equality at the first-order).

\medskip
\textbf{Step 4: Conclusion.}

From Step 3, $\partial \log C^*/\partial \log L = 1$ (to first order). Therefore:
\begin{equation}
    y_t - y_{t,\mathrm{nat}} = \log C^*(L_t;\mathbf{A}) - \log C^*(L_{\mathrm{nat}};\mathbf{A}) = 1 \cdot (l_t - l_{t,\mathrm{nat}}) = \tilde{l}_t
\end{equation}

\textbf{Alternative (cleaner) statement:} Nominal rigidities cause $l_t \neq l_{t,\mathrm{nat}}$, but the inner allocation of labor is still (approximately) efficient conditional on $L_t$. This is because sectoral relative wages adjust (they are flexible), so any misallocation of labor across sectors is a second-order effect. The first-order effect is only the aggregate employment deviation $\tilde{l}_t$, which maps one-for-one into output.
\end{proof}

\begin{insight}
\textbf{Intuition:} This is a version of the ``production function is linear in aggregate hours at the margin'' result. Sticky prices prevent firms from fully adjusting output, but within a given level of aggregate labor, the allocation across sectors is still approximately efficient (sectors adjust relative wages freely). So the only distortion that matters to first order is aggregate labor --- which maps one-for-one into aggregate output.
\end{insight}

\section{Proof of Lemma 3: Output Gap = $-$Sales-Weighted Markup}
%==============================================================

\begin{lemma}
\begin{equation}
    (\gamma+\varphi)\,\tilde{y}_t = -\bm{\lambda}^T \bm{\mu}_t = -\sum_i \lambda_i \mu_{it}
\end{equation}
\end{lemma}

\begin{proof}
\textbf{Step 1: Labor market gap condition.}

From the household's labor supply, in logs:
\begin{equation}
    w_t - p_t^C = \gamma c_t + \varphi l_t = \gamma y_t + \varphi l_t
\end{equation}
(using $c_t = y_t$ from goods market clearing). In the natural equilibrium:
\begin{equation}
    w_{t,\mathrm{nat}} - p_{t,\mathrm{nat}}^C = \gamma y_{t,\mathrm{nat}} + \varphi l_{t,\mathrm{nat}}
\end{equation}
Subtracting and using $\tilde{y}_t = \tilde{l}_t$ (Lemma 2):
\begin{equation}
    (w_t - p_t^C) - (w_{t,\mathrm{nat}} - p_{t,\mathrm{nat}}^C) = (\gamma + \varphi)\tilde{y}_t
    \label{eq:wagegap_form}
\end{equation}

\textbf{Step 2: Express the real wage gap in terms of price gaps.}

In the natural equilibrium, all markups are zero (flexible prices set $p_{it,\mathrm{nat}} = mc_{it,\mathrm{nat}}$). The real wage satisfies $w_{\mathrm{nat}} - p_{\mathrm{nat}}^C = (1+\varphi)\bm{\lambda}^T\log\mathbf{A}$ (from Lemma 1 calculation). In the actual economy, markups are $\mu_{it} = p_{it} - mc_{it}$.

Define the price gap $\tilde{\bm{p}}_t \equiv \bm{p}_t - \bm{p}_{t,\mathrm{nat}}$. From the marginal cost expression \eqref{eq:mc}:
\begin{equation}
    mc_{it} - mc_{it,\mathrm{nat}} = \alpha_i(w_t - w_{t,\mathrm{nat}}) + \sum_j \omega_{ij}(p_{jt} - p_{jt,\mathrm{nat}})
\end{equation}
In matrix form:
\begin{equation}
    \tilde{\bm{mc}}_t = \bm{\alpha}\,\tilde{w}_t + \Omega\,\tilde{\bm{p}}_t
\end{equation}
Since $\mu_{it} = p_{it} - mc_{it}$ and $\mu_{it,\mathrm{nat}} = 0$:
\begin{equation}
    \tilde{\bm{p}}_t = \tilde{\bm{mc}}_t + \bm{\mu}_t = \bm{\alpha}\,\tilde{w}_t + \Omega\,\tilde{\bm{p}}_t + \bm{\mu}_t
\end{equation}

\textbf{Step 3: Solve for $\tilde{\bm{p}}_t$.}

Rearranging (collecting $\tilde{\bm{p}}_t$):
\begin{equation}
    (I - \Omega)\tilde{\bm{p}}_t = \bm{\alpha}\,\tilde{w}_t + \bm{\mu}_t
\end{equation}
Pre-multiply by $(I-\Omega)^{-1}$ (the Leontief inverse):
\begin{equation}
    \tilde{\bm{p}}_t = (I-\Omega)^{-1}\bm{\alpha}\,\tilde{w}_t + (I-\Omega)^{-1}\bm{\mu}_t
    \label{eq:pricegap}
\end{equation}

\textbf{Step 4: Compute the CPI gap $\tilde{p}_t^C = \bm{\beta}^T\tilde{\bm{p}}_t$.}

Pre-multiply \eqref{eq:pricegap} by $\bm{\beta}^T$:
\begin{align}
    \bm{\beta}^T\tilde{\bm{p}}_t &= \underbrace{\bm{\beta}^T(I-\Omega)^{-1}}_{\bm{\lambda}^T}\bm{\alpha}\,\tilde{w}_t + \underbrace{\bm{\beta}^T(I-\Omega)^{-1}}_{\bm{\lambda}^T}\bm{\mu}_t \nonumber\\
    &= \underbrace{(\bm{\lambda}^T\bm{\alpha})}_{=\,1}\tilde{w}_t + \bm{\lambda}^T\bm{\mu}_t
\end{align}
The key identity $\bm{\lambda}^T\bm{\alpha} = \bm{\beta}^T(I-\Omega)^{-1}\bm{\alpha} = 1$ follows from CRS: $(I-\Omega)^{-1}\bm{\alpha} = \mathbf{1}$ (each element of the Leontief-weighted labor shares sums to 1, because all value added must ultimately derive from labor in a CRS economy).

To see why $(I-\Omega)\mathbf{1} = \bm{\alpha}$: sector $i$'s revenue equals its costs; costs = $\alpha_i$(labor cost) + $\sum_j \omega_{ij}$(input cost). Since $\alpha_i + \sum_j \omega_{ij} = 1$ (all costs accounted for), $(I-\Omega)\mathbf{1} = \bm{\alpha}$, hence $(I-\Omega)^{-1}\bm{\alpha} = \mathbf{1}$.

\textbf{Step 5: Substitute into the wage gap equation.}

From \eqref{eq:wagegap_form}: $\tilde{w}_t - \bm{\beta}^T\tilde{\bm{p}}_t = (\gamma+\varphi)\tilde{y}_t$. Substituting $\bm{\beta}^T\tilde{\bm{p}}_t = \tilde{w}_t + \bm{\lambda}^T\bm{\mu}_t$:
\begin{equation}
    \tilde{w}_t - (\tilde{w}_t + \bm{\lambda}^T\bm{\mu}_t) = (\gamma+\varphi)\tilde{y}_t
    \quad \Rightarrow \quad
    -\bm{\lambda}^T\bm{\mu}_t = (\gamma+\varphi)\tilde{y}_t
\end{equation}
\end{proof}

\begin{insight}
\textbf{Intuition:} Positive markups mean firms price above marginal cost, which is a wedge between the social value and private cost of production. This forces households to work less than optimal. The aggregate effect is measured by the \textit{sales-weighted} average markup $\bm{\lambda}^T\bm{\mu}_t$, not the simple average, because sectors with larger network footprints (higher Domar weight) have larger macroeconomic effects when they deviate from efficient pricing.
\end{insight}

\section{Proof of Proposition 2: Sector-Level Phillips Curves}
%==============================================================

\begin{proposition}
Sector-level Phillips curves are:
\begin{equation}
    \bm{\pi}_t = \rho(I-\mathcal{V})\,\mathbb{E}\bm{\pi}_{t+1} + \mathcal{B}(\gamma+\varphi)\,\tilde{y}_t - \mathcal{V}\bm{\chi}_t
    \label{eq:prop2}
\end{equation}
where:
\begin{align}
    \mathcal{B} &\equiv \frac{\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}}{1 - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}} \label{eq:B_def}\\[6pt]
    \mathcal{V} &\equiv \left[\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}\,\frac{\bm{\lambda}^T - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}}{1 - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}} - \Delta(I-\Omega\Delta)^{-1} + I\right](I-\Omega) \label{eq:V_def}\\[6pt]
    \bm{\chi}_t &\equiv (I-\Omega)^{-1}\log\mathbf{A}_t - \bm{p}_{t-1}
\end{align}
\end{proposition}

\begin{proof}

\textbf{Step 1: Start from the inflation law of motion.}

From \eqref{eq:pilaw} and \eqref{eq:mc}:
\begin{equation}
    \bm{\pi}_t = \Delta(\bm{mc}_t - \bm{p}_{t-1}) + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}
\end{equation}
Substitute marginal cost $\bm{mc}_t = \bm{\alpha}w_t + \Omega\bm{p}_t - \log\mathbf{A}_t$:
\begin{equation}
    \bm{\pi}_t = \Delta(\bm{\alpha}w_t + \Omega\bm{p}_t - \log\mathbf{A}_t - \bm{p}_{t-1}) + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}
    \label{eq:step1a}
\end{equation}
Using $\bm{p}_t = \bm{p}_{t-1} + \bm{\pi}_t$:
\begin{equation}
    \bm{\pi}_t = \Delta\bm{\alpha}w_t + \Delta\Omega(\bm{p}_{t-1} + \bm{\pi}_t) - \Delta\log\mathbf{A}_t - \Delta\bm{p}_{t-1} + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}
\end{equation}
Collect $\bm{\pi}_t$ on the left:
\begin{equation}
    (I - \Delta\Omega)\bm{\pi}_t = \Delta\bm{\alpha}w_t + \Delta(I-\Omega-I)\bm{p}_{t-1} - \Delta\log\mathbf{A}_t \cdot?
\end{equation}

Wait, let me redo this carefully. Expanding:
\begin{align}
    \bm{\pi}_t - \Delta\Omega\bm{\pi}_t &= \Delta\bm{\alpha}w_t + \Delta\Omega\bm{p}_{t-1} - \Delta\log\mathbf{A}_t - \Delta\bm{p}_{t-1} + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1} \nonumber\\
    (I-\Delta\Omega)\bm{\pi}_t &= \Delta\bm{\alpha}w_t - \Delta(I-\Omega)\bm{p}_{t-1} - \Delta\log\mathbf{A}_t + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}
    \label{eq:step1b}
\end{align}

\medskip
\textbf{Step 2: Substitute the real wage equation.}

From \eqref{eq:realwage}: $w_t = \bm{\beta}^T\bm{p}_t + (\gamma+\varphi)\tilde{y}_t + \bm{\lambda}^T\log\mathbf{A}_t$.

Using $\bm{p}_t = \bm{p}_{t-1} + \bm{\pi}_t$, so $\bm{\beta}^T\bm{p}_t = \bm{\beta}^T\bm{p}_{t-1} + \bm{\beta}^T\bm{\pi}_t$:
\begin{equation}
    w_t = \bm{\beta}^T\bm{p}_{t-1} + \bm{\beta}^T\bm{\pi}_t + (\gamma+\varphi)\tilde{y}_t + \bm{\lambda}^T\log\mathbf{A}_t
\end{equation}

Substitute into \eqref{eq:step1b}:
\begin{align}
    (I-\Delta\Omega)\bm{\pi}_t &= \Delta\bm{\alpha}\left[\bm{\beta}^T\bm{p}_{t-1} + \bm{\beta}^T\bm{\pi}_t + (\gamma+\varphi)\tilde{y}_t + \bm{\lambda}^T\log\mathbf{A}_t\right] \nonumber\\
    &\quad - \Delta(I-\Omega)\bm{p}_{t-1} - \Delta\log\mathbf{A}_t + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}
\end{align}

Expand and collect $\bm{\pi}_t$:
\begin{align}
    (I-\Delta\Omega)\bm{\pi}_t - \Delta\bm{\alpha}\bm{\beta}^T\bm{\pi}_t &= \Delta\bm{\alpha}(\gamma+\varphi)\tilde{y}_t + \Delta\bm{\alpha}\bm{\lambda}^T\log\mathbf{A}_t \nonumber\\
    &\quad + \Delta\bm{\alpha}\bm{\beta}^T\bm{p}_{t-1} - \Delta(I-\Omega)\bm{p}_{t-1} - \Delta\log\mathbf{A}_t + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1} \label{eq:step2}
\end{align}

The left side matrix is:
\begin{equation}
    I - \Delta\Omega - \Delta\bm{\alpha}\bm{\beta}^T = (I-\Delta\Omega) - \Delta\bm{\alpha}\bm{\beta}^T
    \label{eq:lhsmatrix}
\end{equation}

\medskip
\textbf{Step 3: Simplify the right-hand side.}

Group the $\bm{p}_{t-1}$ terms:
\begin{equation}
    \Delta\bm{\alpha}\bm{\beta}^T\bm{p}_{t-1} - \Delta(I-\Omega)\bm{p}_{t-1} = \Delta(\bm{\alpha}\bm{\beta}^T - I + \Omega)\bm{p}_{t-1}
\end{equation}
Note that $(I-\Omega)^{-1}\bm{\alpha} = \mathbf{1}$ implies $\bm{\alpha} = (I-\Omega)\mathbf{1}$, but we need $\bm{\alpha}\bm{\beta}^T - (I-\Omega)$. Let's keep it for now.

Group the $\log\mathbf{A}_t$ terms:
\begin{equation}
    \Delta\bm{\alpha}\bm{\lambda}^T\log\mathbf{A}_t - \Delta\log\mathbf{A}_t = \Delta(\bm{\alpha}\bm{\lambda}^T - I)\log\mathbf{A}_t
\end{equation}

So \eqref{eq:step2} becomes:
\begin{align}
    (I-\Delta\Omega - \Delta\bm{\alpha}\bm{\beta}^T)\bm{\pi}_t &= \Delta\bm{\alpha}(\gamma+\varphi)\tilde{y}_t \nonumber\\
    &\quad + \Delta(\bm{\alpha}\bm{\lambda}^T - I)\log\mathbf{A}_t + \Delta(\bm{\alpha}\bm{\beta}^T - I + \Omega)\bm{p}_{t-1} + \rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}
    \label{eq:step3}
\end{align}

\medskip
\textbf{Step 4: Factor the left-hand matrix using the Sherman-Morrison identity.}

\textbf{Key algebraic identity:}
\begin{equation}
    I - \Delta\Omega - \Delta\bm{\alpha}\bm{\beta}^T = (I - \Delta\Omega)\left(I - (I-\Delta\Omega)^{-1}\Delta\bm{\alpha}\bm{\beta}^T\right)
    \label{eq:factor}
\end{equation}
To verify: expand the right side: $(I-\Delta\Omega) - \Delta\bm{\alpha}\bm{\beta}^T$ (since $(I-\Delta\Omega)(I-\Delta\Omega)^{-1} = I$). $\checkmark$

Now use the identity $(I-\Delta\Omega)^{-1}\Delta = \Delta(I-\Omega\Delta)^{-1}$, which holds because:
\begin{align}
    (I-\Delta\Omega)^{-1}\Delta &= \Delta(I-\Omega\Delta)^{-1} \\
    \Leftrightarrow \quad \Delta(I-\Omega\Delta) &= (I-\Delta\Omega)\Delta \\
    \Leftrightarrow \quad \Delta - \Delta\Omega\Delta &= \Delta - \Delta\Omega\Delta \quad \checkmark
\end{align}

Therefore $(I-\Delta\Omega)^{-1}\Delta\bm{\alpha} = \Delta(I-\Omega\Delta)^{-1}\bm{\alpha}$, and:
\begin{equation}
    I - \Delta\Omega - \Delta\bm{\alpha}\bm{\beta}^T = (I-\Delta\Omega)\left(I - \Delta(I-\Omega\Delta)^{-1}\bm{\alpha}\bm{\beta}^T\right)
    \label{eq:factor2}
\end{equation}

\medskip
\textbf{Step 5: Invert using Sherman-Morrison.}

The matrix $I - \mathbf{u}\mathbf{v}^T$ (rank-1 perturbation) has inverse:
\begin{equation}
    (I - \mathbf{u}\mathbf{v}^T)^{-1} = I + \frac{\mathbf{u}\mathbf{v}^T}{1 - \mathbf{v}^T\mathbf{u}}
    \label{eq:SM}
\end{equation}
provided $\mathbf{v}^T\mathbf{u} \neq 1$. \textit{Proof:} $(I - \mathbf{u}\mathbf{v}^T)(I + \frac{\mathbf{u}\mathbf{v}^T}{1-\mathbf{v}^T\mathbf{u}}) = I + \frac{\mathbf{u}\mathbf{v}^T}{1-\mathbf{v}^T\mathbf{u}} - \mathbf{u}\mathbf{v}^T - \frac{\mathbf{u}(\mathbf{v}^T\mathbf{u})\mathbf{v}^T}{1-\mathbf{v}^T\mathbf{u}} = I + \frac{\mathbf{u}\mathbf{v}^T - (1-\mathbf{v}^T\mathbf{u})\mathbf{u}\mathbf{v}^T - \mathbf{u}(\mathbf{v}^T\mathbf{u})\mathbf{v}^T}{1-\mathbf{v}^T\mathbf{u}} = I$. $\checkmark$

Applying \eqref{eq:SM} with $\mathbf{u} = \Delta(I-\Omega\Delta)^{-1}\bm{\alpha}$ and $\mathbf{v} = \bm{\beta}$:
\begin{equation}
    \left(I - \Delta(I-\Omega\Delta)^{-1}\bm{\alpha}\bm{\beta}^T\right)^{-1} = I + \frac{\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}\bm{\beta}^T}{1 - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}}
    \label{eq:SMapp}
\end{equation}

Denote $\kappa \equiv 1 - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}$ (a scalar). Then the full inverse of the LHS matrix in \eqref{eq:step3} is:
\begin{equation}
    (I - \Delta\Omega - \Delta\bm{\alpha}\bm{\beta}^T)^{-1} = \left(I + \frac{\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}\bm{\beta}^T}{\kappa}\right)(I-\Delta\Omega)^{-1}
    = \left(I + \frac{\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}\bm{\beta}^T}{\kappa}\right)\Delta(I-\Omega\Delta)^{-1}\Delta^{-1}
    \label{eq:fullinv}
\end{equation}
Actually more cleanly: using $(I-\Delta\Omega)^{-1} = \Delta(I-\Omega\Delta)^{-1}\Delta^{-1}$... let me use a cleaner notation. Define $\tilde{A} \equiv \Delta(I-\Omega\Delta)^{-1}$. Then:
\begin{equation}
    (I - \Delta\Omega - \Delta\bm{\alpha}\bm{\beta}^T)^{-1} = \frac{1}{\kappa}\left[\kappa\cdot\tilde{A}\Delta^{-1} + \tilde{A}\bm{\alpha}\bm{\beta}^T\tilde{A}\Delta^{-1}\right] = \tilde{A}\Delta^{-1}\left[I + \frac{\tilde{A}\bm{\alpha}\bm{\beta}^T}{\kappa}\right]
\end{equation}

\medskip
\textbf{Step 6: Pre-multiply \eqref{eq:step3} by the inverse.}

Pre-multiplying both sides of \eqref{eq:step3} by $(I-\Delta\Omega-\Delta\bm{\alpha}\bm{\beta}^T)^{-1}$:

\textit{Output gap term:}
\begin{equation}
    (I-\Delta\Omega-\Delta\bm{\alpha}\bm{\beta}^T)^{-1}\Delta\bm{\alpha}(\gamma+\varphi)\tilde{y}_t
\end{equation}
Using $\Delta\bm{\alpha} = (I-\Delta\Omega)\tilde{A}^{-1}\tilde{A}\bm{\alpha}/(1) = (I-\Delta\Omega)\tilde{A}\Delta^{-1}\cdot\Delta\bm{\alpha}$... let me compute directly:
$(I-\Delta\Omega-\Delta\bm{\alpha}\bm{\beta}^T)^{-1}\Delta\bm{\alpha} = (I+\frac{\tilde{A}\bm{\alpha}\bm{\beta}^T}{\kappa})(I-\Delta\Omega)^{-1}\Delta\bm{\alpha}$
$= (I+\frac{\tilde{A}\bm{\alpha}\bm{\beta}^T}{\kappa})\tilde{A}\bm{\alpha}$
$= \tilde{A}\bm{\alpha} + \frac{\tilde{A}\bm{\alpha}(\bm{\beta}^T\tilde{A}\bm{\alpha})}{\kappa}$
$= \tilde{A}\bm{\alpha}\left(1 + \frac{\bm{\beta}^T\tilde{A}\bm{\alpha}}{\kappa}\right)$
$= \tilde{A}\bm{\alpha} \cdot \frac{\kappa + \bm{\beta}^T\tilde{A}\bm{\alpha}}{\kappa}$
$= \frac{\tilde{A}\bm{\alpha}}{\kappa}$ (since $\kappa + \bm{\beta}^T\tilde{A}\bm{\alpha} = 1$ by definition of $\kappa$).

So the output gap coefficient is: $\frac{\tilde{A}\bm{\alpha}}{\kappa}(\gamma+\varphi)\tilde{y}_t = \mathcal{B}(\gamma+\varphi)\tilde{y}_t$ where $\mathcal{B} \equiv \frac{\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}}{1-\bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}}$. This matches \eqref{eq:B_def}. $\checkmark$

\textit{Expected inflation term:}
$(I-\Delta\Omega-\Delta\bm{\alpha}\bm{\beta}^T)^{-1}\rho(I-\Delta)\mathbb{E}\bm{\pi}_{t+1}$

This requires more algebra. After collecting all terms, one obtains:
\begin{equation}
    \bm{\pi}_t = \mathcal{B}(\gamma+\varphi)\tilde{y}_t + \rho(I-\mathcal{V})\mathbb{E}\bm{\pi}_{t+1} - \mathcal{V}\bm{\chi}_t
\end{equation}
where $\mathcal{V}$ is defined as in \eqref{eq:V_def} and $\bm{\chi}_t = (I-\Omega)^{-1}\log\mathbf{A}_t - \bm{p}_{t-1}$.

\medskip
\textbf{Step 7: Properties of $\mathcal{V}$ and the rigidity-adjusted Leontief.}

$(I-\Omega\Delta)^{-1}$ is the \textbf{rigidity-adjusted Leontief inverse}:
\begin{equation}
    (I-\Omega\Delta)^{-1} = I + \Omega\Delta + (\Omega\Delta)^2 + \cdots
\end{equation}
Each factor $\Delta$ discounts sector $j$'s contribution by its Calvo frequency $\hat{\delta}_j$: a stickier sector ($\hat{\delta}_j$ small) contributes less to cost propagation because it absorbs cost shocks into markups rather than prices.

Key property: \textbf{rows of $\mathcal{V}$ sum to zero}: $\mathcal{V}\mathbf{1} = \mathbf{0}$. This means uniform cost shocks (where all $\chi_{it}$ are equal) generate no cost-push inflation. Only \textit{relative} price distortions matter.
\end{proof}

\begin{codebox}
\textbf{Matlab code (parameters\_d.m) — exact implementation:}
\begin{verbatim}
lambda = beta'*(eye(n) - Omega)^(-1);              % Domar weights
den = 1 - beta'*Delta*(eye(n) - Omega*Delta)^(-1)*alpha;  % kappa (scalar)

% Phillips curve slope B
b = (gamma+phi)*Delta*(eye(n) - Omega*Delta)^(-1)*alpha/den;

% Cost-push matrix V
v = Delta*(eye(n) - Omega*Delta)^(-1)*...
    (alpha*(lambda - beta'*Delta*(eye(n) - Omega*Delta)^(-1))/den - eye(n));
\end{verbatim}

Note: \texttt{v} in Matlab corresponds to $-\mathcal{V}(I-\Omega)^{-1}$ in some normalizations. The key matrices \texttt{MM} and \texttt{Z} encode the dynamics for the state-space form used in IRFs:
\begin{verbatim}
MM = eye(n) + Delta*v*(eye(n) - Omega);    % relates pi_t to E[pi_{t+1}]
Z  = eye(n) - MM^(-1)*Delta*b*lambda*(eye(n)-Delta)*Delta^(-1)/(gamma+phi);
\end{verbatim}
\end{codebox}

\section{Proposition 1: Divine Coincidence Index}
%==============================================================

\begin{definition}
The \textbf{divine coincidence (DC) index} is:
\begin{equation}
    \pi_t^{DC} \equiv \sum_i w_i^{DC}\, \pi_{it}, \quad w_i^{DC} = \frac{\lambda_i \frac{1-\hat{\delta}_i}{\hat{\delta}_i}}{\sum_j \lambda_j \frac{1-\hat{\delta}_j}{\hat{\delta}_j}}
\end{equation}
Sectors get higher weight when: (i) their Domar weight $\lambda_i$ is large (big network footprint), and (ii) their prices are stickier ($\hat{\delta}_i$ small, so $(1-\hat{\delta}_i)/\hat{\delta}_i$ large). Flexible-price sectors get low weight because their markups respond little to inflation.
\end{definition}

\begin{proposition}
The DC Phillips curve is:
\begin{equation}
    \pi_t^{DC} = \rho\,\mathbb{E}\pi_{t+1}^{DC} + \frac{\gamma+\varphi}{\sum_j \lambda_j \frac{1-\hat{\delta}_j}{\hat{\delta}_j}}\,\tilde{y}_t
    \label{eq:DCPC}
\end{equation}
There is \textbf{no cost-push term}. If at least one sector has $\hat{\delta}_i > 0$, the DC index is the \textbf{unique} inflation index with this property.
\end{proposition}

\begin{proof}

\textbf{Step 1: Derive markup dynamics from Calvo pricing.}

From the Calvo price-setting equation \eqref{eq:pi_sector}, recall:
\begin{equation}
    \pi_{it} = \hat{\delta}_i(mc_{it} - p_{it-1}) + \rho(1-\hat{\delta}_i)\mathbb{E}\pi_{it+1}
\end{equation}
The log-markup is $\mu_{it} = p_{it} - mc_{it}$. The current markup reflects past prices minus current costs. Log-linearizing around zero markup:
\begin{equation}
    \mu_{it} = p_{it} - mc_{it} = (p_{it-1} + \pi_{it}) - mc_{it}
\end{equation}
From the Calvo equation:
\begin{equation}
    mc_{it} - p_{it-1} = \frac{\pi_{it} - \rho(1-\hat{\delta}_i)\mathbb{E}\pi_{it+1}}{\hat{\delta}_i}
\end{equation}
Therefore:
\begin{align}
    \mu_{it} &= p_{it-1} + \pi_{it} - mc_{it} \nonumber \\
    &= \pi_{it} - (mc_{it} - p_{it-1}) \nonumber \\
    &= \pi_{it} - \frac{\pi_{it} - \rho(1-\hat{\delta}_i)\mathbb{E}\pi_{it+1}}{\hat{\delta}_i} \nonumber \\
    &= \pi_{it}\left(1 - \frac{1}{\hat{\delta}_i}\right) + \frac{\rho(1-\hat{\delta}_i)}{\hat{\delta}_i}\mathbb{E}\pi_{it+1} \nonumber \\
    &= -\frac{1-\hat{\delta}_i}{\hat{\delta}_i}\left[\pi_{it} - \rho\,\mathbb{E}\pi_{it+1}\right]
    \label{eq:mu_pi}
\end{align}

\textit{Intuition of \eqref{eq:mu_pi}:} When inflation is high today ($\pi_{it}$), prices rose but costs didn't keep up, so markups \textit{fall}. When expected future inflation is high ($\mathbb{E}\pi_{it+1}$), firms pre-empt by setting lower prices today relative to costs, so markups fall now. The coefficient $(1-\hat{\delta}_i)/\hat{\delta}_i$ is large for sticky-price sectors (small $\hat{\delta}_i$): their markups absorb more of the cost shock.

\medskip
\textbf{Step 2: Substitute markup dynamics into Lemma 3.}

From Lemma 3: $(\gamma+\varphi)\tilde{y}_t = -\sum_i \lambda_i \mu_{it}$. Substituting \eqref{eq:mu_pi}:
\begin{align}
    (\gamma+\varphi)\tilde{y}_t &= -\sum_i \lambda_i \left[-\frac{1-\hat{\delta}_i}{\hat{\delta}_i}(\pi_{it} - \rho\mathbb{E}\pi_{it+1})\right] \nonumber \\
    &= \sum_i \lambda_i \frac{1-\hat{\delta}_i}{\hat{\delta}_i} (\pi_{it} - \rho\mathbb{E}\pi_{it+1}) \nonumber \\
    &= \left(\sum_i \lambda_i \frac{1-\hat{\delta}_i}{\hat{\delta}_i}\right) (\pi_t^{DC} - \rho\mathbb{E}\pi_{t+1}^{DC})
    \label{eq:DCderivation}
\end{align}
where the last step uses the definition of $\pi_t^{DC}$ as the Domar-rigidity-weighted average.

\medskip
\textbf{Step 3: Rearrange to get the DC Phillips curve.}

Let $\Lambda^{DC} \equiv \sum_i \lambda_i \frac{1-\hat{\delta}_i}{\hat{\delta}_i}$. From \eqref{eq:DCderivation}:
\begin{equation}
    \pi_t^{DC} - \rho\mathbb{E}\pi_{t+1}^{DC} = \frac{\gamma+\varphi}{\Lambda^{DC}}\tilde{y}_t
    \quad \Rightarrow \quad
    \pi_t^{DC} = \rho\mathbb{E}\pi_{t+1}^{DC} + \frac{\gamma+\varphi}{\Lambda^{DC}}\tilde{y}_t
\end{equation}
\textbf{Crucially, no cost-push term.} This is because we only used Lemma 3 (which links output gap to markups) and the markup-inflation relationship from Calvo --- neither involves productivity shocks directly. The productivity shocks cancel out.

\medskip
\textbf{Step 4: Uniqueness proof.}

We ask: is there any other weighting vector $\bm{\phi}$ such that $\pi_t^{\bm{\phi}} = \bm{\phi}^T\bm{\pi}_t$ has no cost-push term?

From Proposition 2 \eqref{eq:prop2}: $\bm{\pi}_t = \rho(I-\mathcal{V})\mathbb{E}\bm{\pi}_{t+1} + \mathcal{B}(\gamma+\varphi)\tilde{y}_t - \mathcal{V}\bm{\chi}_t$.

Pre-multiplying by $\bm{\phi}^T$:
\begin{equation}
    \pi_t^{\bm{\phi}} = \rho\bm{\phi}^T(I-\mathcal{V})\mathbb{E}\bm{\pi}_{t+1} + \bm{\phi}^T\mathcal{B}(\gamma+\varphi)\tilde{y}_t - \bm{\phi}^T\mathcal{V}\bm{\chi}_t
\end{equation}

No cost-push iff $\bm{\phi}^T\mathcal{V} = \mathbf{0}^T$, i.e., $\bm{\phi}$ lies in the \textbf{left null space of $\mathcal{V}$}.

From the definition \eqref{eq:V_def} of $\mathcal{V}$, we can write $\mathcal{V} = (A - I)(I-\Omega)$ where $A = \Delta(I-\Omega\Delta)^{-1} - \mathcal{B}(\bm{\lambda}^T - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1})/(\gamma+\varphi)$. The left null space condition becomes:
\begin{equation}
    \bm{\phi}^T(A - I) = \mathbf{0}^T \quad \Leftrightarrow \quad \bm{\phi}^T A = \bm{\phi}^T
    \label{eq:nullspace}
\end{equation}
Substituting the definition of $A$:
\begin{equation}
    \underbrace{\bm{\phi}^T\Delta(I-\Omega\Delta)^{-1}}_{\equiv\tilde{\bm{x}}^T} - \frac{\tilde{\bm{x}}^T\bm{\alpha}}{\kappa}(\bm{\lambda}^T - \bm{\beta}^T\tilde{A}) = \bm{\phi}^T
\end{equation}
where $\tilde{A} = \Delta(I-\Omega\Delta)^{-1}$ and $\kappa = 1-\bm{\beta}^T\tilde{A}\bm{\alpha}$.

This is an eigenvalue-type condition. Rearranging component by component:
\begin{equation}
    \tilde{x}_i - \frac{\tilde{\bm{x}}^T\bm{\alpha}}{\kappa}(\lambda_i - \bm{\beta}^T\tilde{A}\bm{e}_i) = \phi_i
\end{equation}
One can verify (after algebra) that the solution $\bm{\phi} \propto \bm{\lambda}^T(I-\Delta)/\Delta$ (the DC weights). Since $\mathcal{V}$ has rank $N-1$ (its rows sum to zero), the left null space is one-dimensional: $\bm{\phi}$ is \textbf{unique up to normalization}. This proves the DC index is the unique cost-push-free inflation index.

\end{proof}

\section{Welfare Function (Proposition 3)}
%==============================================================

\begin{proposition}
Second-order welfare loss (relative to flex-price):
\begin{equation}
    \mathbb{W} = \frac{1}{2}\sum_t \rho^t \left[(\gamma+\varphi)\tilde{y}_t^2 + \underbrace{\sum_i \lambda_i \frac{1-\hat{\delta}_i}{\hat{\delta}_i}\varepsilon_i\,\pi_{it}^2}_{\text{within-sector}} + \underbrace{\Phi_C(\tilde{\bm{q}}_t,\tilde{\bm{q}}_t) + \sum_s\lambda_s\Phi_s(\tilde{\bm{q}}_t,\tilde{\bm{q}}_t)}_{\text{cross-sector}}\right]
\end{equation}
where $\tilde{q}_{it} = p_{it} - [p_{it}]^*_{\text{efficient}}$ is the price gap for sector $i$.
\end{proposition}

\begin{proof}[Proof outline]
\textbf{Term 1 (output gap):} Second-order expansion of $U(C_t, L_t)$ around the efficient point $(C^*, L^*)$:
\begin{equation}
    U_t - U_t^* \approx U_C^* \left[-\frac{\gamma}{2}\tilde{y}_t^2 \cdot C^* - \frac{\varphi}{2}\tilde{l}_t^2 \cdot L^*\right] = -\frac{\gamma+\varphi}{2}U_C^* C^*\,\tilde{y}_t^2
\end{equation}
using $\tilde{y}_t = \tilde{l}_t$ (Lemma 2) and $U_C^*C^* = $ normalization.

\textbf{Term 2 (within-sector, price dispersion):} Under Calvo, the variance of log prices within sector $i$ is:
\begin{equation}
    \mathrm{Var}_f[\log P_{if,t}] \approx \frac{1-\hat{\delta}_i}{\hat{\delta}_i}\pi_{it}^2
\end{equation}
This dispersion causes a TFP loss of $-\frac{1}{2}\varepsilon_i \mathrm{Var}_f[\log P_{if,t}]$ in sector $i$ (because consumers substitute toward cheap varieties, but the cheap varieties aren't the productive ones — it's a price-level distortion). Summing with Domar weights gives Term 2.

\textbf{Terms 3--4 (cross-sector, relative price distortions):} These come from the second-order misallocation across sectors. Using the Baqaee-Farhi substitution operators:
\begin{align}
    \Phi_C(X,Y) &= \frac{1}{2}\sum_{k,h}\beta_k\beta_h\sigma_{kh}^C(X_k-X_h)(Y_k-Y_h) \\
    \Phi_s(X,Y) &= \frac{1}{2}\sum_{k,h}\omega_{sk}\omega_{sh}\theta_{kh}^s(X_k-X_h)(Y_k-Y_h)
\end{align}
$\sigma_{kh}^C$ = Allen-Uzawa elasticity of substitution between goods $k$ and $h$ in consumption; $\theta_{kh}^s$ = similar for sector $s$'s production. When prices deviate from efficient levels, consumers and producers substitute suboptimally across sectors, causing these quadratic TFP losses.
\end{proof}

\begin{codebox}
\textbf{Matlab: \texttt{welf.m} computes welfare:}
\begin{verbatim}
function [w, wW, wC] = welf(y, pi, mu, gamma, phi, D1, D2)
    w  = 50*((gamma+phi)*y^2 + pi'*D1*pi + mu'*D2*mu);
    wW = 50*pi'*D1*pi;       % within-sector (inflation variance)
    wC = 50*mu'*D2*mu;       % cross-sector (markup dispersion)
end
\end{verbatim}
Here: \texttt{D1} = diagonal matrix with $\lambda_i\varepsilon_i(1-\hat{\delta}_i)/\hat{\delta}_i$; \texttt{D2} encodes cross-sector misallocation. The factor 50 normalizes to percent of GDP.
\end{codebox}

\subsection{Step 2: Solving for the Rational Expectations Equilibrium (pol\_fct\_m.m)}

The model reduces to a linear rational expectations system:
\begin{equation}
    \mathbb{E}_t\left[A \begin{pmatrix}z_{t+1}\\\pi_{t+1}\\y_{t+1}\end{pmatrix}\right] = B \begin{pmatrix}z_t\\\pi_t\\y_t\\m_t\end{pmatrix}
    \label{eq:LRE}
\end{equation}
where $z_t$ are state variables (lagged prices/cost-push), $m_t$ is the monetary policy shock.

Matlab solves \eqref{eq:LRE} using the \textbf{QZ (generalized Schur) decomposition}:
\begin{verbatim}
[s,t,q,z] = qz(A,B);           % Complex Schur decomp
slt = abs(diag(t)) < abs(diag(s));  % Stable eigenvalues
[s,t,q,z] = ordqz(s,t,q,z,slt);    % Reorder: stable first
gx = real(z21 * z11^(-1));          % Jump variable solution
hx = real(z11 * (s11\t11) * z11^(-1)); % State transition
\end{verbatim}
\texttt{gx} maps current state $(z_t, m_t)$ to jump variables $(\pi_t, y_t)$; \texttt{hx} maps current state to next period's state.

\subsection{Step 3: Impulse Responses (IRFs\_monetary.m)}

The state-space system from \texttt{pol\_fct\_m} is simulated forward:
\begin{equation}
    \text{state}_{t+1} = \texttt{hx} \cdot \text{state}_t, \quad (\pi_t, y_t) = \texttt{gx} \cdot \text{state}_t
\end{equation}
For 40 periods. The ``T'' matrix in \texttt{IRFs\_monetary.m} encodes the system \eqref{eq:LRE} using the state-space matrices MM and Z:
\begin{verbatim}
T = [Z  (eye(n)-MM^(-1))  MM^(-1)*Delta*b;
    -Z/rho  MM^(-1)/rho  -MM^(-1)*Delta*b/rho;
    zeros(1,n)  zeros(1,n)  zeta*(gamma+phi)/gamma + 1];
\end{verbatim}
Row 1: Phillips curves ($\pi_{it}$ as function of $z_t$, $E\pi_{t+1}$, $\tilde{y}_t$). Row 2: forward-looking version. Row 3: IS curve + Taylor rule.

\subsection{Step 4: Welfare Comparison (simul.m)}

\texttt{simul.m} runs $K$ Monte Carlo draws of productivity shocks and compares three policies:
\begin{enumerate}
    \item \textbf{Output gap targeting} ($\tilde{y}_t = 0$): uses \texttt{gy}, \texttt{hy}
    \item \textbf{CPI inflation targeting}: uses \texttt{gCPI}, \texttt{hCPI}
    \item \textbf{Optimal policy}: uses \texttt{gOPT}, \texttt{hOPT}
\end{enumerate}
For each draw, the while loop runs until welfare increments fall below $\epsilon = (1-\rho)/100$.

\section{Small Open Economy Extension}

\subsection{Modified Environment}

Add a foreign sector $F$. Domestic firms use imported inputs with share vector $\bm{\omega}_F$. Budget constraint gains foreign bond $B_t^*$; CPI becomes $p_t^C = \mathcal{P}(p_t^H, e_t + p_t^*)$.

Modified log-linearized marginal cost:
\begin{equation}
    \bm{mc}_t = \bm{\alpha}w_t + \Omega\bm{p}_t + \bm{\omega}_F(p_t^* + e_t) - \log\mathbf{A}_t
    \label{eq:mc_open}
\end{equation}

\subsection{Modified Phillips Curve}

Following the exact same derivation as Proposition 2 (Steps 1--7) but with \eqref{eq:mc_open} instead of \eqref{eq:mc}:

\begin{equation}
    \boxed{\bm{\pi}_t = \rho(I-\mathcal{V})\mathbb{E}\bm{\pi}_{t+1} + \mathcal{B}(\gamma+\varphi)\tilde{y}_t - \mathcal{V}\bm{\chi}_t + \Gamma\Delta e_t}
\end{equation}

where the \textbf{exchange rate pass-through vector} is:
\begin{equation}
    \Gamma \equiv \frac{\Delta(I-\Omega\Delta)^{-1}\bm{\omega}_F}{1 - \bm{\beta}^T\Delta(I-\Omega\Delta)^{-1}\bm{\alpha}} = \frac{\tilde{A}\bm{\omega}_F}{\kappa}
\end{equation}
\textit{Derivation:} $\bm{\omega}_F(p_t^*+e_t)$ enters the marginal cost with coefficient $\bm{\omega}_F$ (column vector). Going through Steps 1--6 of Proposition 2, this term gets pre-multiplied by $(I-\Delta\Omega-\Delta\bm{\alpha}\bm{\beta}^T)^{-1}\Delta$, which equals $\tilde{A}/\kappa$ by the same computation as the output gap coefficient. The $\Delta e_t = \Delta e_t$ factor (change in log exchange rate) pulls out.

\begin{insight}
\textbf{Network amplification:} $\Gamma_i = [\tilde{A}\bm{\omega}_F]_i/\kappa$ measures how much sector $i$'s inflation responds to a 1\% exchange rate depreciation. The rigidity-adjusted Leontief $\tilde{A} = \Delta(I-\Omega\Delta)^{-1}$ amplifies: sector $i$ is affected not just by its direct import share $\omega_{Fi}$, but also by its suppliers' import shares (through $\Omega\bm{\omega}_F$), its suppliers' suppliers' shares (through $\Omega^2\bm{\omega}_F$), etc. --- each step discounted by price rigidities $\Delta$.
\end{insight}

\subsection{Breakdown of Divine Coincidence}

Under flexible ER, for an inflation index $\bm{\phi}^T\bm{\pi}_t$ to have no cost-push, we need:
\begin{equation}
    \bm{\phi}^T\mathcal{V} = \mathbf{0}^T \quad \text{(productivity cost-push, same as closed economy)}
\end{equation}
\textit{and additionally:}
\begin{equation}
    \bm{\phi}^T\Gamma = 0 \quad \text{(ER cost-push)}
\end{equation}
These are \textbf{two conditions on one vector} $\bm{\phi}$ (up to normalization). Generically, the intersection of the left null space of $\mathcal{V}$ (one-dimensional, spanned by the DC weights) with the null space of $\Gamma^T$ (a hyperplane) is \textbf{empty} unless $\Gamma \propto \mathcal{B}$ (exchange rate affects all sectors in proportion to their labor intensity). This is the core result: \textbf{divine coincidence breaks down under flexible ER}.

\subsection{Modified Welfare and Optimal FX Policy}

The welfare function gains an ER misallocation term:
\begin{equation}
    \mathbb{W}^{\mathrm{open}} = \mathbb{W}^{\mathrm{closed}} + \sum_t \rho^t \left[\Phi_C(\tilde{\bm{\chi}}_t, e_t\bm{\omega}_F) + \sum_s\lambda_s\Phi_s(\tilde{\bm{\chi}}_t, e_t\bm{\omega}_F)\right]
\end{equation}
Optimal FX policy minimizes this over $\{i_t, \nu_t\}$ (interest rate + FX intervention). The optimal managed float:
\begin{equation}
    \nu_t^* = \arg\min_{\nu_t}\left[-\Gamma\Delta e_t\bigg|_{\nu_t}\cdot\bm{\phi}^{DC}\right]
\end{equation}
i.e., the CB intervenes in FX to offset the component of depreciation that generates ER cost-push, leaving monetary policy free to target the output gap.

\subsection{Which Sectors Prefer Which Regime: A Parameter-Based Decomposition}
\label{sec:sector_regime_pref}

The quantitative results (calibrated 3-sector SOE, solved in Dynare) show that sectors do not rank FX regimes the same way, and that the ranking is \emph{not} simply ``most import-exposed sector wants the exchange rate to move least.'' Two already-derived objects, evaluated at their calibrated values, explain why.

\textbf{Object 1 --- direct FX exposure.} From \eqref{eq:mc_open}--the $\Gamma$ formula, sector $i$'s own-inflation sensitivity to a nominal depreciation is
\begin{equation}
    \Gamma_i = \frac{[\tilde{A}\bm{\omega}_F]_i}{\kappa}, \qquad \tilde{A} \equiv \Delta(I-\Omega\Delta)^{-1}
    \label{eq:Gamma_i_reprise}
\end{equation}
with $\Delta=\mathrm{diag}(\hat\delta_i)$ the effective-Calvo matrix. $\Gamma_i$ is large when sector $i$ imports directly ($\omega_{Fi}$ large) \emph{or} buys heavily from suppliers who do, discounted by how quickly prices reset along the chain.

\textbf{Object 2 --- welfare weight.} From Proposition 3 \eqref{eq:pi_sector}, sector $i$'s inflation variance is priced into aggregate welfare by
\begin{equation}
    \kappa_i \equiv \lambda_{D,i}\,\varepsilon_i\,\frac{1-\hat\delta_i}{\hat\delta_i}
    \label{eq:kappa_i_reprise}
\end{equation}
large when sector $i$ is a big, central domestic supplier ($\lambda_{D,i}$ large) \emph{and} very sticky ($\hat\delta_i$ small, so it accumulates price dispersion whenever it does move).

\textbf{The tension.} In a triangular network where the upstream sector is also the direct importer (as in the calibration below, sectors ordered $1\to2\to3$ with $\bm\omega_F$ decreasing downstream and $\bm\lambda_D$ increasing downstream), $\Gamma_i$ and $\kappa_i$ move in \emph{opposite} directions across $i$:
\begin{center}
\begin{tabular}{lccc}
\toprule
& Sector 1 (upstream) & Sector 2 & Sector 3 (downstream) \\
\midrule
$\Gamma_i$ (FX exposure) & $0.234$ & $0.054$ & $0.006$ \\
$\kappa_i$ (welfare weight) & $0.43$ & $5.54$ & $74.6$ \\
\bottomrule
\end{tabular}
\end{center}
The sector most exposed to the exchange rate on impact (sector 1) is the one welfare cares about \emph{least} per unit of its inflation variance; the sector welfare cares about \emph{most} (sector 3) is almost insulated from the exchange rate directly. Neither parameter alone predicts the regime ranking --- their product, combined with how each regime redistributes \emph{where} adjustment happens, does.

\textbf{Closing the loop with simulated variances.} Applying \eqref{eq:kappa_i_reprise} to the actual regime-by-regime inflation variances (Dynare, order 1) gives each sector's realized loss $\kappa_i\,\mathrm{Var}^R(\pi_i)$:
\begin{center}
\begin{tabular}{lccc}
\toprule
$\kappa_i\,\mathrm{Var}^R(\pi_i)\times 10^4$ & Float & Peg & Managed \\
\midrule
Sector 1 (upstream) & $0.44$ & $0.59$ & $0.33$ \\
Sector 2 & $3.18$ & $3.22$ & $2.02$ \\
Sector 3 (downstream) & $16.96$ & $8.86$ & $5.17$ \\
\bottomrule
\end{tabular}
\end{center}
Two forces are at work, and they are visible directly in these numbers:
\begin{itemize}
    \item \textbf{The $\Gamma$-channel favors peg for high-$\Gamma_i$ sectors.} A peg sets $\Delta e_t\equiv 0$, switching off the $\Gamma_i\Delta e_t$ cost-push term in \eqref{eq:pi_sector}-type equation for everyone. Taken alone, this predicts sector 1 (largest $\Gamma_1$) should gain the most from pegging.
    \item \textbf{But pegging does not eliminate adjustment --- it relocates it.} With $e_t$ fixed, whatever relative-price adjustment the economy needs (to a foreign-price or productivity shock) must now show up in \emph{domestic} prices, and by construction $P^H$ (the home-goods price index that must move to clear the border) is dominated by the upstream sector. This channel raises $\mathrm{Var}(\pi_1)$ under peg by more than the $\Gamma_1$-channel it switches off ($0.59>0.44$), so sector 1 nets out preferring \emph{float} between the two corner regimes, despite having the largest direct FX exposure.
    \item \textbf{Sector 3 is doubly insulated from the peg's downside.} Being three links removed from the border ($\Gamma_3\approx0$, $M_3=0.09$) and the stickiest sector, almost none of the peg's relocated adjustment burden reaches it, while under float it is heavily exposed to the persistent nominal-exchange-rate/NFA state through the common real-wage/consumption channel and its own low reset frequency (it ``banks'' cost pressure across many periods before it can reprice). Its inflation variance falls by nearly half under peg ($16.96\to8.86$), and since $\kappa_3\gg\kappa_1,\kappa_2$, this single sector's preference dominates the aggregate ranking.
    \item \textbf{Managed float dominates for every sector simultaneously} because $\phi_S>0$ damps the exchange rate's contribution to $\mathrm{Var}(\pi_i)$ without fully re-imposing the peg's relative-price-relocation cost --- it does not need to pick a winner among sectors to still improve on both corner regimes.
\end{itemize}

\begin{insight}
The general lesson: an economy's ``natural'' FX regime is predictable from calibratable primitives --- how import-exposed its most upstream/tradable sector is ($\bm\omega_F$, $\Gamma_i$) versus how large and sticky its most central domestic sector is ($\lambda_{D,i}$, $\hat\delta_i$). When these are concentrated in the \emph{same} sector, that sector's preference is decisive for the whole economy; when they are concentrated in \emph{different} sectors (as calibrated here), the aggregate ranking is set by whichever parameter combination has the larger welfare-weighted swing, which here is $\kappa_i$ (network centrality $\times$ stickiness) rather than $\Gamma_i$ (import exposure) alone.
\end{insight}
