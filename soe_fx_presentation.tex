\section{Model}
%==============================================================
\begin{frame}{Model: Households}
%==============================================================

{\scriptsize
\[
\max\ \mathbb{E}_0\sum_t \beta^t\Big[\tfrac{C_t^{1-\gamma}}{1-\gamma} - \tfrac{L_t^{1+\varphi}}{1+\varphi}\Big]
\qquad\text{s.t.}\qquad
P^C_tC_t + S_tB^*_t = W_tL_t + \Pi_t + (1{+}i^*)\big[1{-}\psi(\hat b^*_{t-1}{-}\bar b^*)\big]S_tB^*_{t-1}
\]
\vspace{-8pt}
\[
C_t^{-\gamma} = \beta(1{+}i_t)\mathbb{E}_t\big[C_{t+1}^{-\gamma}/\Pi^C_{t+1}\big],\qquad
W_t/P^C_t = C_t^{\gamma}L_t^{\varphi}
\]
\vspace{-6pt}
\[
\text{UIP (derived):}\quad (1{+}i_t) = (1{+}i^*)\big[1{-}\psi(\hat b^*_t{-}\bar b^*)\big]\,\mathbb{E}_t[S_{t+1}/S_t]
\]}

\vspace{6pt}
{\tiny
\begin{tabular}{@{}llll@{}}
$C,L$: consumption, labor & $\beta,\gamma,\varphi$: discount, RRA, inv.\ Frisch & $W,i$: wage, dom.\ rate & $S,i^*$: ER, foreign rate \\
$\psi$: debt-elastic premium & $B^*$: foreign bond & $P^C$: CPI & $\Pi_t$: profits (rebated) \\
\end{tabular}}

\end{frame}

%==============================================================
\begin{frame}{Model: Firms --- Technology and Cost}
%==============================================================

{\scriptsize
\[
Y_{it}=A_{it}\,L_{it}^{\alpha_i}\prod_j X_{ijt}^{\Omega^H_{ij}}\,M_{it}^{\Omega^F_i}
\qquad\Rightarrow\qquad
MC_{it}=\frac{W_t^{\alpha_i}\prod_j P_{jt}^{\Omega^H_{ij}}\,(S_tP^{F*}_t)^{\Omega^F_i}}{A_{it}}
\]
\vspace{-6pt}
\[
W_tL_{it}=\alpha_i MC_{it}Y_{it},\qquad P_{jt}X_{ijt}=\Omega^H_{ij}MC_{it}Y_{it},\qquad S_tP^{F*}_tM_{it}=\Omega^F_i MC_{it}Y_{it}
\]}

\vspace{6pt}
{\tiny
\begin{tabular}{@{}llll@{}}
$Y,A$: output, TFP & $L,X,M$: labor, dom./imp.\ inputs & $\Omega^H_{ij}$: dom.\ IO share & $\Omega^F_i$: import cost share \\
$\alpha_i$: labor share & $MC_{it}$: nominal marg.\ cost & \multicolumn{2}{l}{$P^{F*}$: foreign import price} \\
\end{tabular}}

\vspace{6pt}
{\footnotesize \textbf{Key:} $MC_{it}$ moves with both $A_{it}$ (TFP) and $S_t$ (via $M_{it}$) --- the two cost-push channels.}

\end{frame}

%==============================================================
\begin{frame}{Model: Firms --- Calvo Pricing}
%==============================================================

{\scriptsize \textbf{Reset price} (probability $\delta_i$ resets, $1{-}\delta_i$ keeps last price):}
\vspace{-4pt}
{\scriptsize
\[
X1_{it} = \widetilde{MC}_{it}Y_{it} + (1{-}\delta_i)\beta\Big(\tfrac{C_{t+1}}{C_t}\Big)^{-\gamma}\Pi_{i,t+1}^{\varepsilon}X1_{i,t+1},\qquad
X2_{it} = Y_{it} + (1{-}\delta_i)\beta\Big(\tfrac{C_{t+1}}{C_t}\Big)^{-\gamma}\Pi_{i,t+1}^{\varepsilon-1}X2_{i,t+1}
\]
\vspace{-8pt}
\[
P_{it}^\#/P_{it} = \tfrac{\varepsilon}{\varepsilon-1}\,X1_{it}/X2_{it},\qquad
1 = (1{-}\delta_i)\Pi_{it}^{\varepsilon-1} + \delta_i \big(P_{it}^\#/P_{it}\big)^{1-\varepsilon}
\]}

\vspace{6pt}
{\tiny
\begin{tabular}{@{}llll@{}}
$\delta_i$: Calvo reset prob.\ (primitive) & $\hat\delta_i$: derived slope & $P_{it}^\#$: reset price & $\varepsilon$: elasticity (markup $\varepsilon/(\varepsilon{-}1)$) \\
\end{tabular}}

\vspace{6pt}
{\footnotesize $\hat\delta_i=\dfrac{\delta_i(1-\beta(1-\delta_i))}{1-\beta\delta_i(1-\delta_i)}$ --- appears in DC weights (Theory section), \emph{derived} not primitive.}

\end{frame}

%==============================================================
\begin{frame}{Model: Open Economy}
%==============================================================

{\scriptsize
\[
\underbrace{P^{FH}_t = S_t P^{F*}_t}_{\text{LOP}},\qquad
\underbrace{(1{+}i_t) = (1{+}i^*)\big[1{-}\psi(\hat b^*_t{-}\bar b^*)\big]\,\mathbb{E}_t[S_{t+1}/S_t]}_{\text{UIP}}
\]
\vspace{-8pt}
\[
\hat b^*_t = \tfrac{1+i_{t-1}}{\Pi^C_t}\hat b^*_{t-1} + \tfrac{P^H_t EX_t - P^{FH}_t IM_t}{P^C_t},\qquad
EX_t = \kappa_{EX}D^*_t\Big(\tfrac{P^H_t}{P^{FH}_t}\Big)^{-\theta^*}
\]
\vspace{-8pt}
\[
Y_{it} = \beta^H_i\tfrac{P^H_t}{P_{it}}\big(C^H_t+EX_t\big) + \sum_j \Omega^H_{ji}\tfrac{MC_{jt}Y_{jt}}{P_{it}}
\]}

\vspace{6pt}
{\tiny
\begin{tabular}{@{}llll@{}}
$\hat b^*$: NFA (scaled) & $EX,IM$: exports, imports & $\theta^*$: export elasticity & $D^*$: foreign demand shifter \\
$\beta^H_i$: cons.\ share, sector $i$ & \multicolumn{3}{l}{Float: $i_t$ Taylor rule, $S_t$ free \quad Peg: $S_t=\bar S$, $i_t$ residual} \\
\end{tabular}}

\end{frame}

%==============================================================
\begin{frame}{Model: Policy Regimes and Shocks}
%==============================================================

{\footnotesize
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Regime} & \textbf{Rule} & \textbf{Exchange rate} \\
\midrule
Float & $i_t = \bar\imath + \phi_\pi\pi^{DC}_t + \phi_y\tilde y_t$ & fully endogenous (UIP) \\
Peg & $S_t = \bar S$ & fixed; $i_t$ residual \\
Managed & $i_t = \bar\imath + \phi_\pi\pi^{DC}_t + \phi_y\tilde y_t + \phi_s\log(S_t/\bar S)$ & partially stabilized \\
\bottomrule
\end{tabular}}

\vspace{10pt}
{\footnotesize \textbf{Shocks} (AR(1)): $a_{it}$ (TFP, $\rho{=}0.90$)\quad $p^{F*}_t$ (import price, $\rho{=}0.85$)\quad $D^*_t$ (foreign demand, $\rho{=}0.80$)}

\end{frame}
