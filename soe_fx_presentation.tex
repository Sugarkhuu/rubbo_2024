\section{Intro}
%==============================================================
\begin{frame}{Motivation}
%==============================================================

{\footnotesize A depreciation raises the cost of imported inputs. But firms don't just buy imports directly --- they buy from \emph{other domestic firms}, who themselves buy imports. A single exchange-rate shock therefore travels through the production network before it ever reaches the consumer: \textbf{the same shock can be small or large, depending on where in the network it lands.}}

\vspace{8pt}
\begin{center}

\vspace{6pt}
{\footnotesize
\begin{itemize}\setlength{\itemsep}{2pt}
\item Network Phillips curves (Rubbo 2024, La'O--Tahbaz-Salehi 2022): get this propagation right --- but \textbf{closed economy}, no exchange rate
\item Open-economy NK (Gal\'i--Monacelli 2005): get the exchange rate right --- but \textbf{no network}, one representative firm
\item \textbf{This paper: both together} --- and asks the question that requires having both
\end{itemize}}

\vspace{6pt}
\begin{center}
{\footnotesize\textbf{Does accounting for production networks change the optimal exchange-rate regime?}}
\end{center}

\end{frame}

%==============================================================
\begin{frame}{Why Production Networks Matter}
%==============================================================

{\footnotesize
\begin{itemize}\setlength{\itemsep}{7pt}
\item \textbf{Intermediate inputs dominate production.} In most economies, firms spend more buying from \emph{other domestic firms} than they spend on labor --- a cost shock to any one sector mechanically becomes a cost shock to everyone downstream of it
\item \textbf{Imported intermediates are the norm, not the exception.} Manufacturing sectors routinely source $15$--$30\%$ of inputs from abroad (Chile: up to $19.5\%$, Manufacturing) --- so an exchange-rate move is, from the firm's perspective, indistinguishable from a domestic cost shock
\item \textbf{Recent shocks propagated through exactly this channel.} COVID-19, the Ukraine war, and the semiconductor shortage each started as a shock to one link and were amplified --- or dampened --- by where that link sat in the network
\end{itemize}}

\vspace{10pt}
{\scriptsize Acemoglu, Carvalho, Ozdaglar \& Tahbaz-Salehi (2012); Baqaee \& Farhi (2019); La'O \& Tahbaz-Salehi (2022)}

\end{frame}

\begin{frame}{Literature}
%==============================================================

{\scriptsize \textbf{Network side (closed economy):}}
{\tiny
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\topsep}{1pt}
\item \textbf{Rubbo (2024)} --- Calvo network $\Rightarrow$ unique DC index (1 cost-push channel)
\item \textbf{La'O \& Tahbaz-Salehi (2022)} --- optimal MP in production networks
\item \textbf{Pasten, Schoenle \& Weber (2020)} --- sectoral rigidity heterogeneity drives MP transmission
\end{itemize}}

{\scriptsize \textbf{Open-economy side (no network):}}
{\tiny
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\topsep}{1pt}
\item \textbf{Gal\'i \& Monacelli (2005)} --- 1-sector SOE: domestic-inflation targeting $\approx$ optimal, peg dominated (ToT channel)
\item \textbf{Fanelli \& Straub (2021)} --- FX intervention as a \emph{second} instrument, separate from $i_t$
\item \textbf{Schmitt-Grohé \& Uribe (2003)} --- debt-elastic premium to stationarize NFA; flags 1st-order welfare risk
\item \textbf{Gopinath \& Itskhoki (2022); Amiti et al.\ (2019); Auer, Levchenko \& Sauré (2019)} --- pass-through, dominant currency, IO spillovers
\item \textbf{Itskhoki \& Mukhin (2023); Bianchi \& Coulibaly (2025); Coulibaly (2023)} --- frontier: managed float/fear-of-floating \emph{optimal}, not behavioral (no network)
\end{itemize}}

{\scriptsize \textbf{Open economy \emph{with} network (closest to this paper):}}
{\tiny
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\topsep}{1pt}
\item \textbf{Gnocato, Montes-Gald\'on \& Stamato (2025, ECB); Kalemli-\"Ozcan et al.\ (2025)} --- open economy \emph{with} network (tariffs), fixed Taylor rule, no FX-regime choice\footnote{\tiny Details \& what these papers cite for open-economy production networks: \hyperlink{ecbnetwork}{Appendix}.}
\item \textbf{Qiu, Wang, Xu \& Zanetti (2026, JME)} --- closest paper: derives the open-economy DC index directly extending Rubbo, WIOD 43-country calibration; but \emph{static} (no NFA/persistence), no UIP, and no FX-regime choice --- their own conclusion lists relaxing financial autarky as their top open extension
\end{itemize}}

{\scriptsize \textbf{Gap:} network $+$ open economy $+$ FX \textbf{regime choice} together --- no one has this. Same ranking as Gal\'i--Monacelli but a \textbf{different channel} (risk-premium, not ToT); Managed's dominance echoes Fanelli--Straub \& Itskhoki--Mukhin, but from network persistence.}

\section{Model}
%==============================================================
%% my slides

%------------------------------------------------
\begin{frame}{Households}

\small

Small open economy: home takes foreign price $P_t^{F*}$, rate $i^*$, and demand $D_t^*$ as given.\footnote{\tiny Full equations \& notation for every block in this section: \hyperlink{modeleqns}{Appendix}.}

\vspace{0.15cm}

Representative household chooses $\{C_t,L_t,B_t,B_t^*\}$ to maximize CRRA-consumption/disutility-of-labor utility, subject to a budget constraint with a domestic bond $B_t$ (nets to zero in equilibrium) and a foreign bond $B_t^*$:

\[
E_0\sum_{t=0}^{\infty}\beta^t
\left[
\frac{C_t^{1-\gamma}}{1-\gamma}
-
\frac{L_t^{1+\varphi}}{1+\varphi}
\right]
\]
\[
P_t^CC_t+B_t+S_tB_t^*=W_tL_t+\Pi_t+(1{+}i_{t-1})B_{t-1}+(1{+}i^*)\big[1-\psi(\hat b^*_{t-1}-\bar b^*)\big]S_tB_{t-1}^*
\]

Consumption is CES across home/foreign goods, Cobb-Douglas across domestic sectors:

\[
C_t=
\left[
\omega^{1/\eta}(C_t^H)^{\frac{\eta-1}{\eta}}
+
(1-\omega)^{1/\eta}(C_t^F)^{\frac{\eta-1}{\eta}}
\right]^{\frac{\eta}{\eta-1}},
\qquad
C_t^H=\prod_i(C_{it}^H)^{\beta_i^H}
\]

\end{frame}


%------------------------------------------------
\begin{frame}{Firms: Technology}

\small

Firms in sector $i$ share Cobb-Douglas technology (labor, domestic inputs, imports) and price under monopolistic competition; CRS makes marginal cost common across firms:

\[
Y_{ift}=A_{it}\,L_{ift}^{\alpha_i}\prod_j X_{ijft}^{\Omega^H_{ij}}\,M_{ift}^{\Omega^F_i}
\]
\[
MC_{it}=\frac{W_t^{\alpha_i}\prod_j P_{jt}^{\Omega^H_{ij}}\,(S_tP_t^{F*})^{\Omega^F_i}}{A_{it}}
\]

\textbf{Two cost-push channels:} $MC_{it}$ moves with $A_{it}$ (TFP) \emph{and} $S_t$ (import cost); with \textbf{Calvo pricing}, only a fraction $\delta_i$ resets each period --- mechanics \& Phillips curve: next slide.

\end{frame}


%------------------------------------------------
\begin{frame}{Production Network}

\small

Stacking each firm's input shares $\Omega^H_{ij}$, $\Omega^F_i$ across sectors gives the \textbf{input-output network} $\Omega^H\in\mathbb R^{N\times N}$ and import-exposure vector $\bm\omega_F\in\mathbb R^N$.

\vspace{0.15cm}

The \textbf{Leontief inverse} sums direct and indirect linkages:

\[
(I-\Omega^H)^{-1}=I+\Omega^H+(\Omega^H)^2+\cdots
\]

applied to consumption shares $\bm\beta^H$ (Households) it gives each sector's \textbf{Domar weight} (total GDP sales share); applied to $\bm\omega_F$, its total foreign exposure:

\[
\bm\lambda^\top=(\bm\beta^H)^\top(I-\Omega^H)^{-1},
\qquad
\mathcal M=(I-\Omega^H)^{-1}\bm\omega_F
\]

\end{frame}

%------------------------------------------------
\begin{frame}{Firms: Price Setting}

\footnotesize

Calvo pricing through the network delivers the (vector) Phillips curve:

\[
\bm\pi_t=\rho(I-\mathcal V)\mathbb E_t\bm\pi_{t+1}+\mathcal B(\gamma+\varphi)\tilde y_t-\mathcal V\bm\chi_t+\Gamma\Delta e_t
\]

\textbf{Same amplifier, two shocks:} $\mathcal B,\Gamma$ are the \emph{same} rigidity-adjusted operator $\tilde A\equiv\Delta(I-\Omega^H\Delta)^{-1}$, applied to labor share $\bm\alpha$ vs.\ import share $\bm\omega_F$:

\[
\mathcal B=\frac{\tilde A\bm\alpha}{\kappa},
\qquad
\Gamma=\frac{\tilde A\bm\omega_F}{\kappa}
\]

\textbf{Only relative shocks matter:} $\mathcal V\mathbf 1=\mathbf 0$ --- uniform TFP shocks create no cost-push; only sector-specific gaps $\bm\chi_t=(I-\Omega^H)^{-1}\log\mathbf A_t-\bm p_{t-1}$ do.

{\footnotesize where:}

{\scriptsize
\begin{tabular}{@{}llll@{}}
$\bm\pi_t,\tilde y_t$: infl., gap & $\bm\chi_t$: TFP cost-push & $\Delta e_t$: exch.\ rate chg.\ & $\kappa$: GE scalar \\
$\mathcal B,\Gamma$: PC slope, FX pass-thru\ & $\mathcal V$: TFP cost-push matrix & \multicolumn{2}{l}{$\tilde A$: rigidity-adj.\ Leontief} \\
\end{tabular}}

\end{frame}


%------------------------------------------------
\begin{frame}{Foreign Sector}

\footnotesize

$P_t^{F*}$, $i^*$, $D_t^*$ are exogenous AR(1) processes. Law of one price:

\[
P_t^{FH}=S_tP_t^{F*}
\]

UIP, with a debt-elastic risk premium closing the model (Schmitt-Grohé--Uribe) and $RP_t$ the risk-premium shock studied below:

\[
i_t = i^*+\mathbb E_t\Delta e_{t+1}+\psi(\hat b_t^*)+RP_t
\]

NFA accumulates the trade balance; exports are \textbf{sector-specific}, each responding to its own relative price (baseline model --- not a single aggregate $EX_t$ allocated by consumption shares):

\[
\hat b_t^*=\frac{1+i_{t-1}}{\Pi_t^C}\hat b_{t-1}^*+\frac{P_t^H\sum_iEX_{it}-P_t^{FH}IM_t}{P_t^C},
\qquad
EX_{it}=\kappa_{EX,i}D_t^*\Big(\frac{P_{it}}{P_t^{FH}}\Big)^{-\theta^*}
\]

$\kappa_{EX,i}$ is calibrated so each sector's steady-state export share matches Chile's real data (Resource-heavy: $0.602/0.180/0.036$) --- not the Services-heavy household shares $\beta_i^H$ that a single aggregate $EX_t$ would imply.

\textbf{Two channels} to the domestic economy:

\[
\hat s_t+\hat p_t^{F*}\ \rightarrow\ MC_{it}\ \rightarrow\ \pi_{it}
\quad\text{(cost-push)}
\qquad
\hat s_t\ \rightarrow\ EX_{it}\ \rightarrow\ \tilde y_t\ \rightarrow\ \pi_{it}
\quad\text{(demand)}
\]

{\scriptsize\textbf{Shocks (7 total, all AR(1), std.\ dev.\ 0.01):} 3 sector TFP $\varepsilon^A_{1,2,3}$ ($\rho{=}0.90$) $\cdot$ import price $\varepsilon^{P^{F*}}$ ($\rho{=}0.85$, into $P_t^{F*}$) $\cdot$ foreign demand $\varepsilon^{D^*}$ ($\rho{=}0.80$) $\cdot$ export price/ToT $\varepsilon^{PX}$ ($\rho{=}0.85$) $\cdot$ risk premium $\varepsilon^{RP}$ ($\rho{=}0.80$, into $RP_t$ above) --- decomposed by contribution to welfare loss in Results.}

\end{frame}



%------------------------------------------------
\begin{frame}{Monetary Policy Regimes}

\small

Three exchange-rate regimes are compared. All target the \textbf{DC index $\pi_t^{DC}$, not CPI inflation}: Rubbo (2024) shows CPI targeting does \emph{not} achieve divine coincidence even in the closed economy --- $\pi_t^{DC}$ is the closed-economy-optimal benchmark, so it is the natural rule to test for survival once the economy is opened.

\vspace{0.3cm}

\textbf{Free float}

\[
\hat i_t
=
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t
\]

The exchange rate adjusts endogenously.

\vspace{0.3cm}

\textbf{Hard peg}

\[
\hat s_t=0
\]

Adjustment occurs through domestic inflation and output.

\vspace{0.3cm}

\textbf{Managed float}

\[
\hat i_t
=
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\hat s_t
\]

The central bank partially stabilizes the exchange rate.

\end{frame}


%------------------------------------------------
\begin{frame}{Equilibrium System}

\footnotesize

Everything above collapses to four blocks --- this is what is actually coded up and simulated in Dynare.

\textbf{Phillips curve} (vector, network-amplified, two cost-push channels):

\[
\bm\pi_t=\rho(I-\mathcal V)\mathbb E_t\bm\pi_{t+1}+\mathcal B(\gamma+\varphi)\tilde y_t-\mathcal V\bm\chi_t+\Gamma\Delta e_t
\]

\textbf{IS curve} (aggregate demand; open economy adds the same relative-price term as the cost-push channel above):

\[
\tilde y_t=\mathbb E_t\tilde y_{t+1}-\frac{1}{\gamma}\big(i_t-\mathbb E_t\pi_t^C-r_t^{\mathrm{nat}}\big)-\frac{\eta(1-\omega)}{\gamma}\mathbb E_t\Delta\big(\hat s_t+\hat p_t^{F*}-\hat p_t^H\big)
\]

\textbf{Foreign sector} (UIP closes the FX block; NFA is the only state carried across regimes):

\[
i_t=i^*+\mathbb E_t\Delta e_{t+1}+\psi(\hat b_t^*)+RP_t,
\qquad
\hat b_t^*=\frac{1+i_{t-1}}{\Pi_t^C}\hat b_{t-1}^*+\frac{P_t^H\sum_iEX_{it}-P_t^{FH}IM_t}{P_t^C}
\]

Exports are sector-specific, $EX_{it}=\kappa_{EX,i}D_t^*(P_{it}/P_t^{FH})^{-\theta^*}$ (baseline; $\kappa_{EX,i}$ calibrated to Chile's real export shares).

\textbf{Policy rule} (regime-dependent; the only block that changes across Float/Peg/Managed --- same rules as previous slide):

\[
\hat i_t=
\begin{cases}
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t & \text{Float}\\[2pt]
\text{residual (}\hat s_t=0\text{)} & \text{Peg}\\[2pt]
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\hat s_t & \text{Managed}
\end{cases}
\]

{\scriptsize where:}

{\scriptsize
\begin{tabular}{@{}lll@{}}
$r_t^{\mathrm{nat}}$: natural real rate & $\hat p_t^H$: log home price idx.\ & $\omega,\eta$: home bias, elast.\ \\
$\psi(\cdot),RP_t$: debt wedge, risk-prem.\ & $\pi_t^{DC}$: DC index (target) & $\mathcal B,\Gamma,\mathcal V$: PC coefficients \\
\end{tabular}}

\end{frame}

\item \textbf{Magnitudes/ranking --- need simulation:} $\mathrm{Var}(\pi_{it}),\mathrm{Var}(\tilde y_t)$ come from the full linear RE solution (order-1 Dynare) --- no closed form once $\phi_S{>}0$ or $\Omega^H$ dense. \end{itemize}}

{\scriptsize \textbf{Closed economy (Rubbo):} 1 channel (TFP), $\mathcal V$ rank-deficient $\Rightarrow$ unique DC index:}
\vspace{-3pt}
{\scriptsize\[ \bm\phi^\top\mathcal{V}=\bm 0^\top \Rightarrow \pi^{DC}_t = \textstyle\sum_i w^{DC}_i\pi_{it},\ w^{DC}_i\propto\lambda_{D,i}\tfrac{1-\hat\delta_i}{\hat\delta_i}\]}

{\scriptsize \textbf{Open economy: 2 channels} (TFP $+$ FX):}
\vspace{-3pt}
{\scriptsize\[ \underbrace{\bm\phi^\top\mathcal{V}=\bm 0^\top}_{\text{TFP}} \text{ and } \underbrace{\bm\phi^\top\Gamma = 0}_{\text{FX}} \Rightarrow \textbf{generically infeasible}\]}

\vspace{-2pt}
{\scriptsize \textbf{Knife-edge test:} $\Gamma\propto\mathcal{B}$ would save the DC index. Instead: $\cos(\Gamma,\mathcal{B})\approx0.96$ ($\approx17^\circ$ apart) --- close, but $\Gamma_i/\mathcal{B}_i$ shrinks $\approx$2$\times$ per sector, not constant. Breakdown is real, not a numerical artifact.
\textbf{Fix:} import nodes flexible ($\hat\delta_M{=}1\Rightarrow w_M{=}0$) $\Rightarrow$ DC well-defined again by construction --- what the Taylor rules target throughout ($\pi^{DC}_t$); FX insulation itself is a quantitative question, shown below.}

{\scriptsize\textbf{Households}}
\vspace{-2pt}
{\tiny
\[
\max\ \mathbb{E}_0\sum_t \beta^t\Big[\tfrac{C_t^{1-\gamma}}{1-\gamma} - \tfrac{L_t^{1+\varphi}}{1+\varphi}\Big]
\quad\text{s.t.}\quad
P^C_tC_t + S_tB^*_t = W_tL_t + \Pi_t + (1{+}i^*)\big[1{-}\psi(\hat b^*_{t-1}{-}\bar b^*)\big]S_tB^*_{t-1}
\]
\vspace{-8pt}
\[
C_t^{-\gamma} = \beta(1{+}i_t)\mathbb{E}_t\big[C_{t+1}^{-\gamma}/\Pi^C_{t+1}\big],\quad
W_t/P^C_t = C_t^{\gamma}L_t^{\varphi},\quad
(1{+}i_t) = (1{+}i^*)\big[1{-}\psi(\hat b^*_t{-}\bar b^*)\big]\,\mathbb{E}_t[S_{t+1}/S_t]\ \text{(UIP)}
\]}
{\tiny
\begin{tabular}{@{}llll@{}}
$C,L$: consumption, labor & $\beta,\gamma,\varphi$: discount, RRA, inv.\ Frisch & $W,i$: wage, dom.\ rate & $S,i^*$: ER, foreign rate \\
$\psi$: debt-elastic premium & $B^*$: foreign bond & $P^C$: CPI & $\Pi_t$: profits (rebated) \\
\end{tabular}}

\vspace{10pt}
{\scriptsize\textbf{Firms: Technology and Cost}}
\vspace{-2pt}
{\tiny
\[
Y_{it}=A_{it}\,L_{it}^{\alpha_i}\prod_j X_{ijt}^{\Omega^H_{ij}}\,M_{it}^{\Omega^F_i}
\quad\Rightarrow\quad
MC_{it}=\frac{W_t^{\alpha_i}\prod_j P_{jt}^{\Omega^H_{ij}}\,(S_tP^{F*}_t)^{\Omega^F_i}}{A_{it}}
\]
\vspace{-6pt}
\[
W_tL_{it}=\alpha_i MC_{it}Y_{it},\quad P_{jt}X_{ijt}=\Omega^H_{ij}MC_{it}Y_{it},\quad S_tP^{F*}_tM_{it}=\Omega^F_i MC_{it}Y_{it}
\]}
{\tiny
\begin{tabular}{@{}llll@{}}
$Y,A$: output, TFP & $L,X,M$: labor, dom./imp.\ inputs & $\Omega^H_{ij}$: dom.\ IO share & $\Omega^F_i$: import cost share \\
$\alpha_i$: labor share & $MC_{it}$: nominal marg.\ cost & \multicolumn{2}{l}{$P^{F*}$: foreign import price} \\
\end{tabular}}

\vspace{6pt}
{\footnotesize \textbf{Key:} $MC_{it}$ moves with both $A_{it}$ (TFP) and $S_t$ (via $M_{it}$) --- the two cost-push channels.}

\section*{Appendix: Extended Literature Review (Backup)}
%==============================================================
\begin{frame}{Appendix: The Peg vs.\ Float Debate --- What Do We Know?}
\hypertarget{littfull}{}
%==============================================================

{\footnotesize \textbf{Why it depends} (structure):}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{1pt}
\item \textbf{Mundell (1961)} OCA --- fix favored by small, open, factor-mobile economies w/ correlated shocks vs.\ anchor; float favored by large, diversified ones
\item \textbf{Frankel (1999)} --- ``no single regime is right for all countries or at all times''; openness, size, trade \& capital-market structure all matter
\end{itemize}}

\vspace{2pt}
{\footnotesize \textbf{Shock-absorption view} (why float wins on real shocks --- robustly):}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{1pt}
\item \textbf{Friedman (1953)} --- $S_t$ adjusts instead of costly wage/price deflation
\item \textbf{Broda (2004)} --- 75 dev.\ countries, 1973--96: negative ToT shocks $\to$ large real GDP declines under pegs
\item \textbf{Edwards \& Levy-Yeyati (2005)} --- 10\% ToT deterioration: $-0.80$pp growth under pegs vs.\ $-0.43$pp under floats
\item \textbf{Céspedes, Chang \& Velasco (2004)} --- even with dollarized-liability balance-sheet effects, float \emph{still} beats fixed
\end{itemize}}

\vspace{2pt}
{\footnotesize \textbf{But floating is not a free lunch} (limits on float's insulation):}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{1pt}
\item \textbf{Calvo \& Reinhart (2002)} ``Fear of Floating'' --- EMEs peg de facto anyway: credibility-building, not literal welfare optimum
\item \textbf{Rey (2013)} ``Dilemma not trilemma'' --- under free capital mobility, float alone doesn't deliver monetary autonomy
\end{itemize}}

\vspace{3pt}
{\scriptsize \textbf{Consensus:} no corner solution is universally optimal --- the choice trades a nominal anchor against real-shock absorption, shock- and structure-dependent.}

\end{frame}

%==============================================================
\begin{frame}{Appendix: The Frontier --- Is Fear of Floating Actually Optimal?}
%==============================================================

{\scriptsize
\begin{itemize}\setlength{\itemsep}{5pt}
\item \textbf{Itskhoki \& Mukhin (2023)} --- general open-economy framework with endogenous UIP/PPP deviations: efficient policy $=$ MP closes the output gap $+$ FX intervention eliminates the UIP wedge. Optimal policy is generically a \textbf{managed float/crawling peg}, matching observed fear of floating. When the natural RER is stable, this collapses to an ``open-economy divine coincidence.''
\\ \textcolor{customgreen!55!black}{$\hookrightarrow$ \emph{vs.\ us:}} \textbf{same headline} (Managed, not a corner, is optimal) --- reached instead via a multi-sector Calvo network with a persistence-driven, not UIP-gap-closing, mechanism

\item \textbf{Bianchi \& Coulibaly (2025; 2022 WP)} --- fear of floating is \textbf{optimal}, not behavioral, when credit access is tied to the real exchange rate (self-fulfilling crisis risk). Prescription: float for real shocks, \textbf{fix} for financial shocks
\\ \textcolor{customgreen!55!black}{$\hookrightarrow$ \emph{vs.\ us:}} \textbf{opposite ranking} for our financial shock --- our risk-premium/UIP shock passes straight through under \textbf{Peg} (worst regime). Different financial friction (persistent NFA wedge vs.\ collateral-constraint crisis) flips the prescription

\item \textbf{Coulibaly (2023)} --- discretionary MP is procyclical to offset balance-sheet effects; commitment (inflation targeting) or capital controls raise welfare
\\ \textcolor{customgreen!55!black}{$\hookrightarrow$ \emph{vs.\ us:}} echoes our ``Managed $=$ extra instrument'' logic --- giving policy more tools beats a corner regime
\end{itemize}}

\vspace{4pt}
{\scriptsize \textbf{Bottom line:} the 2023--25 frontier has also moved past corner solutions --- but \emph{why} fear of floating is optimal is mechanism-specific. We add production-network amplification $+$ shock persistence as a distinct reason, not examined by either paper.}

\end{frame}

{\scriptsize \textbf{Gnocato, Montes-Gald\'on \& Stamato (2025)}, ``Tariffs across the supply chain,'' ECB WP No.\ 3081 --- also underlies Lane's (ECB Exec.\ Board) 13 May 2026 speech on energy shocks (4-region extension):}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{1pt}
\item \textbf{Setup:} two-region, multi-sector, full IO structure, international production network; sticky wages/prices; Taylor rule on \emph{domestic} CPI (no regime comparison)
\item \textbf{Finding:} tariffs on \textbf{intermediate} inputs $\to$ persistent inflation \& larger GDP loss (low substitutability, network propagation); tariffs on \textbf{final} goods $\to$ one-off inflation spike, smaller GDP loss --- \emph{where} in the network a shock hits sets its persistence
\item \textcolor{customgreen!55!black}{$\hookrightarrow$ \emph{vs.\ us:}} their tariff-location result and our TFP-vs-UIP result are the same insight in two settings: shock \textbf{location/type} sets its persistence, echoing our ``driver is persistence, not volatility''
\end{itemize}}

\vspace{2pt}
{\scriptsize \textbf{What it cites for open economy $+$ production networks:}}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{1pt}
\item Network amplification, closed economy: \textbf{Pasten, Schoenle \& Weber (2020)}; \textbf{Ghassibe (2021)}; \textbf{Rubbo (2023)}\footnote{\tiny Published version: Rubbo, E. (2023), Econometrica 91(4), 1417--1455 --- journal year is 2023, not 2024.}; \textbf{Afrouzi \& Bhattarai (2023)}

\end{itemize}}
