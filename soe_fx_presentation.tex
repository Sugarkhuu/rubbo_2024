\vspace{8pt}
{\small
\begin{itemize}\setlength{\itemsep}{5pt}
\item Firms buy inputs mostly from \emph{other domestic firms}, not just labor --- a cost shock to one sector mechanically hits everyone downstream
\item Imported inputs are the norm (Chile: up to $19.5\%$ in Manufacturing) --- so an FX move is, for the firm, a domestic cost shock
\item Network models get this propagation right but ignore the exchange rate; open-economy models get the exchange rate right but ignore the network
\end{itemize}}

\vspace{8pt}
\begin{center}
{\small\textbf{Does accounting for production networks change the optimal exchange-rate regime?}}
\end{center}

\section{Model}
%==============================================================
%% my slides

%------------------------------------------------
\begin{frame}{Households}

\small

Representative household chooses $\{C_t,L_t,B_t,B_t^*\}$ to maximize lifetime utility, subject to a budget constraint with a domestic bond $B_t$ and a foreign bond $B_t^*$:

\[
E_0\sum_{t=0}^{\infty}\beta^t
\left[
\frac{C_t^{1-\gamma}}{1-\gamma}
-
\frac{L_t^{1+\varphi}}{1+\varphi}
\right]
\]

subject to 
\vspace{1pt}

\[
P_t^CC_t+B_t+S_tB_t^*=W_tL_t+\Pi_t+(1{+}i_{t-1})B_{t-1}+(1{+}i^*)\big[1-\psi(\hat b^*_{t-1}-\bar b^*)\big]S_tB_{t-1}^*
\]

\vspace{6pt}
Consumption is CES across home/foreign goods, Cobb-Douglas across domestic sectors:
\[
C_t=
\left[
\omega^{1/\eta}(C_t^H)^{\frac{\eta-1}{\eta}}
+
(1-\omega)^{1/\eta}(C_t^F)^{\frac{\eta-1}{\eta}}
\right]^{\frac{\eta}{\eta-1}},
\qquad
C_t^H=\prod_i(C_{it}^H)^{\beta_i^H}
\]

\end{frame}


%------------------------------------------------
\begin{frame}{Firms: Technology}

\small

Firms in sector $i$ share Cobb-Douglas technology (labor, domestic inputs, imports) and price under monopolistic competition:

\[
Y_{ift}=A_{it}\,L_{ift}^{\alpha_i}\prod_j X_{ijft}^{\Omega^H_{ij}}\,M_{ift}^{\Omega^F_i}
\]

\vspace{1pt}

\[
MC_{it}=\frac{W_t^{\alpha_i}\prod_j P_{jt}^{\Omega^H_{ij}}\,(S_tP_t^{F*})^{\Omega^F_i}}{A_{it}}
\]


\vspace{6pt}
\textbf{Two cost-push channels:} $MC_{it}$ moves with $A_{it}$ (TFP) \emph{and} $S_t$, the exchange rate (via the import cost $S_tP_t^{F*}$).

\vspace{6pt}
The theory holds for any number of sectors $N$; the quantitative results use $N=3$ (Resource, Manufacturing, Services), collapsed from Chile's 12-sector national IO table.

\end{frame}


%------------------------------------------------
\begin{frame}{Firms: Price Setting}

\footnotesize

Calvo pricing through the network delivers the (vector) Phillips curve:

\vspace{6pt}
\[
\bm\pi_t=\rho(I-\mathcal V)\mathbb E_t\bm\pi_{t+1}+\mathcal B(\gamma+\varphi)\tilde y_t-\mathcal V\bm\chi_t+\Gamma\Delta e_t
\]
\vspace{6pt}

\textbf{Same amplifier, two shocks:} $\mathcal B,\Gamma$ are the \emph{same} rigidity-adjusted operator $\tilde A\equiv\Delta(I-\Omega^H\Delta)^{-1}$, applied to labor share $\bm\alpha$ vs.\ import share $\bm\omega_F$:

\vspace{6pt}
\[
\mathcal B=\frac{\tilde A\bm\alpha}{\kappa},
\qquad
\Gamma=\frac{\tilde A\bm\omega_F}{\kappa}
\]


\vspace{10pt}
{\footnotesize where:}

\vspace{4pt}
{\scriptsize
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}llll@{}}
$\bm\pi_t,\tilde y_t$: infl., gap & $\bm\chi_t$: TFP cost-push & $\Delta e_t$: exch.\ rate chg.\ & $\kappa$: GE scalar \\
$\mathcal B,\Gamma$: PC slope, FX pass-thru\ & $\mathcal V$: TFP cost-push matrix & \multicolumn{2}{l}{$\tilde A$: rigidity-adj.\ Leontief} \\
\end{tabular}}

\end{frame}

%------------------------------------------------
\begin{frame}{Firms: Production Network}

\small

The \textbf{Leontief inverse} sums direct and indirect linkages:

\[
(I-\Omega^H)^{-1}=I+\Omega^H+(\Omega^H)^2+\cdots
\]

applied to consumption shares $\bm\beta^H$ (Households) it gives:

\[
\bm\lambda_D^\top=(\bm\beta^H)^\top(I-\Omega^H)^{-1},
\qquad
\mathcal M=(I-\Omega^H)^{-1}\bm\omega_F,
\qquad
\mathcal M^X=(I-\Omega^H)^{-1}\bm\omega_X
\]

\vspace{6pt}
{\footnotesize where:}

\vspace{2pt}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{1pt}\setlength{\topsep}{0pt}
\item $\lambda_{D,i}$: \textbf{Domar weight} --- sector $i$'s total weight in final demand, network-adjusted (can exceed 1 for downstream sectors)
\item $\mathcal M_i$: \textbf{Import centrality} --- sector $i$'s total (direct $+$ indirect) exposure to import costs
\item $\mathcal M_i^X$: \textbf{Export centrality} --- sector $i$'s total (direct $+$ indirect) exposure to export demand
\end{itemize}}

\end{frame}


%------------------------------------------------
\begin{frame}{Foreign Sector}

\footnotesize

Law of one price and UIP with a debt-elastic NFA stabilizer $\psi(\cdot)$ and an exogenous risk-premium shock $RP_t$:

\vspace{10pt}
\[
P_t^{FH}=S_tP_t^{F*},
\qquad
i_t = i^*+\mathbb E_t\Delta e_{t+1}+\psi(\hat b_t^*)+RP_t
\]
\vspace{10pt}

NFA accumulates the trade balance:

\vspace{10pt}
\[
\hat b_t^*=\frac{1+i_{t-1}}{\Pi_t^C}\hat b_{t-1}^*+\frac{P_t^H\sum_iEX_{it}-P_t^{FH}IM_t}{P_t^C},
\qquad
EX_{it}=\kappa_{EX,i}D_t^*\,PX_t\Big(\frac{P_{it}}{P_t^{FH}}\Big)^{-\theta^*}
\]
\vspace{10pt}

\textbf{Two channels} to the domestic economy:

\vspace{10pt}
\[
\hat s_t+\hat p_t^{F*}\ \rightarrow\ MC_{it}\ \rightarrow\ \pi_{it}
\quad\text{(cost-push)}
\qquad
\hat s_t\ \rightarrow\ EX_{it}\ \rightarrow\ \tilde y_t\ \rightarrow\ \pi_{it}
\quad\text{(demand)}
\]

\vspace{6pt}
{\footnotesize where:}

\vspace{4pt}
\resizebox{\textwidth}{!}{%
{\scriptsize
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}llll@{}}
$P_t^{F*}$: foreign price of imports & $RP_t$: risk-premium shock & $D_t^*$: foreign demand shock & $PX_t$: export price/ToT shock \\
$\psi(\cdot)$: debt-elastic NFA stabilizer & $\hat b_t^*$: NFA (\% GDP) & $\kappa_{EX,i}$: export scale, sector $i$ & $\theta^*$: export demand elasticity \\
$\hat s_t\equiv\log(S_t/\bar S)$: log exch.\ rate & $\Delta e_{t+1}\equiv\mathbb E_t\hat s_{t+1}-\hat s_t$: exp.\ depreciation & \multicolumn{2}{l}{(hats: log-deviation from steady state, throughout)} \\
\end{tabular}}}

\end{frame}



%------------------------------------------------
\begin{frame}{Monetary Policy Regimes}

\small


\vspace{10pt}

\textbf{Free float}

\[
\hat i_t
=
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t
\]

The exchange rate adjusts endogenously.

\vspace{10pt}

\textbf{Hard peg}

\[
\hat s_t=0
\]

Adjustment occurs through domestic inflation and output.

\vspace{10pt}

\textbf{Managed float}

\[
\hat i_t
=
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\hat s_t
\]

The central bank partially stabilizes the exchange rate.

\end{frame}


%------------------------------------------------
\begin{frame}{Equilibrium System}

\footnotesize
\setlength{\abovedisplayskip}{1pt}
\setlength{\belowdisplayskip}{1pt}
\setlength{\abovedisplayshortskip}{0pt}
\setlength{\belowdisplayshortskip}{1pt}
\setlength{\parskip}{0pt}

\textbf{Phillips curve}:

\[
\bm\pi_t=\rho(I-\mathcal V)\mathbb E_t\bm\pi_{t+1}+\mathcal B(\gamma+\varphi)\tilde y_t-\mathcal V\bm\chi_t+\Gamma\Delta e_t
\]

\textbf{IS curve}:

\[
\tilde y_t=\mathbb E_t\tilde y_{t+1}-\frac{1}{\gamma}\big(i_t-\mathbb E_t\pi_t^C-r_t^{\mathrm{nat}}\big)-\frac{\eta(1-\omega)}{\gamma}\mathbb E_t\Delta\big(\hat s_t+\hat p_t^{F*}-\hat p_t^H\big)
\]

\textbf{Foreign sector}:

\[
i_t=i^*+\mathbb E_t\Delta e_{t+1}+\psi(\hat b_t^*)+RP_t
\]

\textbf{Policy rule}:

\[
\hat i_t=
\begin{cases}
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t & \text{Float}\\[2pt]
\text{residual (}\hat s_t=0\text{)} & \text{Peg}\\[2pt]
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\hat s_t & \text{Managed}
\end{cases}
\]

% {\scriptsize where: $r_t^{\mathrm{nat}}$ natural real rate, $\hat p_t^H$ log home price idx., $\pi_t^{DC}$ DC index (target); $\mathcal B,\Gamma,\mathcal V,\psi(\cdot),RP_t,\hat p_t^{F*}(PF_t)$ as before.}

% \textbf{Welfare loss}: $\mathcal{W}=\tfrac12\big[(\gamma+\varphi)\mathrm{Var}(\tilde y_t)+\sum_i\kappa_i\mathrm{Var}(\pi_{it})\big]$, $\kappa_i=\lambda_{D,i}\varepsilon\frac{1-\hat\delta_i}{\hat\delta_i}$

\vspace{9pt}
{\scriptsize \textbf{Seven AR(1) shocks} (1\% s.d.\ each): sector TFP $A_1,A_2,A_3$ ($\rho{=}0.90$), import price $PF$ ($\rho{=}0.85$), foreign demand $D^*$ ($\rho{=}0.80$), export price/ToT $PX$ ($\rho{=}0.85$), risk premium $RP$ ($\rho{=}0.80$).}

\end{frame}

\begin{frame}{Isolating the Network Channel}
%==============================================================

{\footnotesize Does the network's effect scale \emph{continuously} with how interconnected the economy is?}

\vspace{4pt}
\begin{center}
\includegraphics[width=0.62\textwidth]{figs/isolating_network.pdf}
\end{center}

\vspace{4pt}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{2pt}\setlength{\topsep}{0pt}
% \item Scales all six cross-sector $\Omega^H$ entries by $\rho$, own-input fixed ($\rho{=}1$ baseline) --- \textbf{Managed stays best throughout}
\item Denser network amplifies shock propagation for \emph{everyone}, but \textbf{Peg} has no policy lever ($\phi_\pi,\phi_y=0$) to lean against it, so the amplification passes straight into welfare loss; \textbf{Float}/\textbf{Managed} actively damp part of it with the interest rate --- \textbf{gap widens} as $\rho\uparrow$
\end{itemize}}

\end{frame}

%==============================================================
\begin{frame}{Does the Optimal Managed-Float Response Depend on the Network?}
%==============================================================

{\footnotesize Christian's question: welfare loss changes a lot with network density --- does the \textbf{welfare-minimizing} FX-stabilization coefficient $\phi_s$ change too, or just its magnitude?}

\vspace{4pt}
\begin{center}
\includegraphics[width=0.72\textwidth]{figs/phi_s_netdens.pdf}
\end{center}

\vspace{2pt}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{2pt}\setlength{\topsep}{0pt}
\item Re-solves the managed-float rule $\hat\imath_t=\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\log S_t$ over the $\phi_s$ grid at three network densities: \textbf{no network} ($\rho{=}0$), \textbf{baseline} ($\rho{=}1$), \textbf{denser} ($\rho{=}2$)
\item \textbf{$\phi_s^*$ rises with density: $0.15\to0.20\to0.30$} --- denser network, more indirect cost-push (previous slide), so the welfare-minimizing rule leans harder against the exchange rate
\item \textbf{But the curve is flat near its minimum} in every case (within $\approx$2--3\% of $\phi_s^*$ across $\phi_s\in[0.10,0.30]$) --- the shift is real and monotonic, not a knife-edge, but getting $\phi_s$ within $\pm0.1$ of "right" matters far more than getting it exactly right
\end{itemize}}

\end{frame}

%==============================================================
\begin{frame}{How the Network Amplifies Shocks}
%==============================================================

{\footnotesize The network \textcolor{fxgreen}{\textbf{dampens}} the inflation peak but \textcolor{riskred}{\textbf{amplifies}} the output-gap peak --- in \emph{every} regime, for \emph{both} shocks. }

\vspace{5pt}
\begin{center}
{\scriptsize
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{@{}llcc@{}}
\toprule
\textbf{Shock} & \textbf{Regime} & \textbf{Inflation peak} & \textbf{Output-gap peak} \\
 & & \multicolumn{2}{c}{(Network ON $\div$ Network OFF)} \\
\midrule
\multirow{3}{*}{Import-price} & Float   & \textcolor{fxgreen}{$0.40\times$} & \textcolor{riskred}{$2.77\times$} \\
 & Managed & \textcolor{fxgreen}{$0.61\times$} & \textcolor{riskred}{$2.28\times$} \\
 & Peg     & \textcolor{fxgreen}{$0.86\times$} & \textcolor{riskred}{$\mathbf{3.15\times}$ } \\
\midrule

\begin{frame}{Why Services? Direct vs.\ Indirect Import Exposure}
%==============================================================

{\footnotesize Services is the \textbf{stickiest} sector (Calvo reset prob.\ $\delta_3{=}0.16$) and has the \textbf{lowest direct} import share ($\Omega^F_3{=}0.070$) --- yet it buys $5.8\%$ of its cost from Manufacturing, the most import-intensive sector ($\Omega^F_2{=}0.195$).}

\vspace{4pt}
\begin{center}
\includegraphics[width=0.86\textwidth]{figs/import_exposure_decomposition.pdf}
\end{center}

\vspace{2pt}
{\scriptsize
\begin{itemize}\setlength{\itemsep}{2pt}\setlength{\topsep}{0pt}
\item Total import centrality $M_i=[(I-\Omega^H)^{-1}\Omega^F\mathbf 1]_i$ splits into a \textbf{direct} part (own $\Omega^F_i$) and an \textbf{indirect} part inherited from suppliers --- for Services, \textbf{41\% of total exposure is indirect} (Resource 50\%, Manuf.\ 32\%)
\item Indirect exposure is \textbf{mechanically zero} at $\rho{=}0$ and grows with network density $\rho$ --- density doesn't change any sector's \emph{direct} import share, only how much of \emph{other} sectors' import content gets passed to it
\item \textbf{Combined with stickiness}: Services can't reprice quickly, so whatever cost-push it inherits shows up as persistent inflation, not a one-quarter blip --- $\mathrm{Corr}(\pi_3,\pi_3^{t-1})\approx0.82$, the most persistent inflation series in the model
\end{itemize}}

\end{frame}

\section{Conclusion}
%==============================================================
\begin{frame}[plain]{Conclusion}
%==============================================================

{\footnotesize
\begin{itemize}\setlength{\itemsep}{6pt}
\item \textbf{The network matters for severity and incidence, not the ranking:} switching it on raises welfare losses in every regime and reallocates who bears them, but \textbf{Managed $<$ Float $<$ Peg either way}
% \item \textbf{Peg is dominated by a different mechanism than the textbook story:} not terms-of-trade, but risk-premium/UIP --- zero monetary autonomy forces the shock straight into the output gap
\item \textbf{Managed float wins} because it dampens the FX pass-through channel without fully reimposing peg's loss of autonomy
\item \textbf{Robust} across three real-data calibrations (Chile, Korea, Czechia) and a genuine second-order solution
\end{itemize}}

\end{frame}
