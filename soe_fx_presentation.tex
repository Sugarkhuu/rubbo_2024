\vspace{8pt}
{\small
\begin{itemize}\setlength{\itemsep}{5pt}
\item Firms buy inputs mostly from \emph{other domestic firms}, not just labor --- a cost shock to one sector mechanically hits everyone downstream
\item Imported inputs are the norm (Chile: up to $19.5\%$ in Manufacturing) --- so an FX move is, for the firm, a domestic cost shock
\item Network models get this propagation right but ignore the exchange rate; open-economy models get the exchange rate right but ignore the network
\item \textbf{This paper: both together}
\end{itemize}}

\vspace{8pt}
\begin{center}
{\small\textbf{Does accounting for production networks change the optimal exchange-rate regime?}}
\end{center}

\end{frame}

\section{Model}
%==============================================================
%% my slides

%------------------------------------------------
\begin{frame}{Households}

\small

Small open economy: home takes foreign price $P_t^{F*}$, rate $i^*$, and demand $D_t^*$ as given.

\vspace{0.15cm}

Representative household chooses $\{C_t,L_t,B_t,B_t^*\}$ to maximize lifetime utility, subject to a budget constraint with a domestic bond $B_t$ and a foreign bond $B_t^*$:

\[
E_0\sum_{t=0}^{\infty}\beta^t
\left[
\frac{C_t^{1-\gamma}}{1-\gamma}
-
\frac{L_t^{1+\varphi}}{1+\varphi}
\right]
\]
\[
P_t^CC_t+B_t+S_tB_t^*=W_tL_t+\Pi_t+(1{+}i_{t-1})B_{t-1}+(1{+}i^*)\big[1-\psi(\hat b^*_{t-1}-\bar b^*)\big]S_tB_{t-1}^*
\]

Consumption is CES across home/foreign goods, Cobb-Douglas across domestic sectors:

\[
C_t=
\left[
\omega^{1/\eta}(C_t^H)^{\frac{\eta-1}{\eta}}
+
(1-\omega)^{1/\eta}(C_t^F)^{\frac{\eta-1}{\eta}}
\right]^{\frac{\eta}{\eta-1}},
\qquad
C_t^H=\prod_i(C_{it}^H)^{\beta_i^H}
\]

\end{frame}


%------------------------------------------------
\begin{frame}{Firms: Technology}

\small

Firms in sector $i$ share Cobb-Douglas technology (labor, domestic inputs, imports) and price under monopolistic competition; CRS makes marginal cost common across firms:

\[
Y_{ift}=A_{it}\,L_{ift}^{\alpha_i}\prod_j X_{ijft}^{\Omega^H_{ij}}\,M_{ift}^{\Omega^F_i}
\]
\[
MC_{it}=\frac{W_t^{\alpha_i}\prod_j P_{jt}^{\Omega^H_{ij}}\,(S_tP_t^{F*})^{\Omega^F_i}}{A_{it}}
\]

\textbf{Two cost-push channels:} $MC_{it}$ moves with $A_{it}$ (TFP) \emph{and} $S_t$, the exchange rate (via the import cost $S_tP_t^{F*}$); with \textbf{Calvo pricing}, only a fraction $\delta_i$ resets each period.

\end{frame}


%------------------------------------------------
\begin{frame}{Firms:Production Network}

\small

Stacking each firm's input shares $\Omega^H_{ij}$, $\Omega^F_i$ across sectors gives the \textbf{input-output network} $\Omega^H\in\mathbb R^{N\times N}$, import-exposure vector $\bm\omega_F\in\mathbb R^N$ (import cost share of each sector's output), and export-exposure vector $\bm\omega_X\in\mathbb R^N$ (export share of each sector's own output).

\vspace{0.15cm}

The \textbf{Leontief inverse} sums direct and indirect linkages:

\[
(I-\Omega^H)^{-1}=I+\Omega^H+(\Omega^H)^2+\cdots
\]

applied to consumption shares $\bm\beta^H$ (Households) it gives each sector's \textbf{Domar weight}; applied to $\bm\omega_F$ or $\bm\omega_X$, its import or export centrality:

\[
\bm\lambda_D^\top=(\bm\beta^H)^\top(I-\Omega^H)^{-1},
\qquad
\mathcal M=(I-\Omega^H)^{-1}\bm\omega_F,
\qquad
\mathcal M^X=(I-\Omega^H)^{-1}\bm\omega_X
\]

\end{frame}

%------------------------------------------------
\begin{frame}{Firms: Price Setting}

\footnotesize

Calvo pricing through the network delivers the (vector) Phillips curve:

\vspace{6pt}
\[
\bm\pi_t=\rho(I-\mathcal V)\mathbb E_t\bm\pi_{t+1}+\mathcal B(\gamma+\varphi)\tilde y_t-\mathcal V\bm\chi_t+\Gamma\Delta e_t
\]
\vspace{6pt}

\textbf{Same amplifier, two shocks:} $\mathcal B,\Gamma$ are the \emph{same} rigidity-adjusted operator $\tilde A\equiv\Delta(I-\Omega^H\Delta)^{-1}$, applied to labor share $\bm\alpha$ vs.\ import share $\bm\omega_F$:

\vspace{6pt}
\[
\mathcal B=\frac{\tilde A\bm\alpha}{\kappa},
\qquad
\Gamma=\frac{\tilde A\bm\omega_F}{\kappa}
\]
\vspace{6pt}

\textbf{Only relative shocks matter:} $\mathcal V\mathbf 1=\mathbf 0$ --- uniform TFP shocks create no cost-push; only sector-specific gaps $\bm\chi_t=(I-\Omega^H)^{-1}\log\mathbf A_t-\bm p_{t-1}$ do.

\vspace{10pt}
{\footnotesize where:}

\vspace{4pt}
{\scriptsize
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}llll@{}}
$\bm\pi_t,\tilde y_t$: infl., gap & $\bm\chi_t$: TFP cost-push & $\Delta e_t$: exch.\ rate chg.\ & $\kappa$: GE scalar \\
$\mathcal B,\Gamma$: PC slope, FX pass-thru\ & $\mathcal V$: TFP cost-push matrix & \multicolumn{2}{l}{$\tilde A$: rigidity-adj.\ Leontief} \\
\end{tabular}}

\end{frame}


%------------------------------------------------
\begin{frame}{Foreign Sector}

\footnotesize

Law of one price and UIP with a debt-elastic risk premium $RP_t$:

\vspace{4pt}
\[
P_t^{FH}=S_tP_t^{F*},
\qquad
i_t = i^*+\mathbb E_t\Delta e_{t+1}+\psi(\hat b_t^*)+RP_t
\]
\vspace{4pt}

NFA accumulates the trade balance; exports are \textbf{sector-specific} (Resource-heavy, matching Chile's data):

\vspace{4pt}
\[
\hat b_t^*=\frac{1+i_{t-1}}{\Pi_t^C}\hat b_{t-1}^*+\frac{P_t^H\sum_iEX_{it}-P_t^{FH}IM_t}{P_t^C},
\qquad
EX_{it}=\kappa_{EX,i}D_t^*\Big(\frac{P_{it}}{P_t^{FH}}\Big)^{-\theta^*}
\]
\vspace{4pt}

\textbf{Two channels} to the domestic economy:

\vspace{4pt}
\[
\hat s_t+\hat p_t^{F*}\ \rightarrow\ MC_{it}\ \rightarrow\ \pi_{it}
\quad\text{(cost-push)}
\qquad
\hat s_t\ \rightarrow\ EX_{it}\ \rightarrow\ \tilde y_t\ \rightarrow\ \pi_{it}
\quad\text{(demand)}
\]
\vspace{4pt}


\end{frame}



%------------------------------------------------
\begin{frame}{Monetary Policy Regimes}

\small

Three exchange-rate regimes are compared. 

\vspace{0.3cm}

\textbf{Free float}

\[
\hat i_t
=
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t
\]

The exchange rate adjusts endogenously.

\vspace{0.3cm}

\textbf{Hard peg}

\[
\hat s_t=0
\]

Adjustment occurs through domestic inflation and output.

\vspace{0.3cm}

\textbf{Managed float}

\[
\hat i_t
=
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\hat s_t
\]

The central bank partially stabilizes the exchange rate.

\end{frame}


%------------------------------------------------
\begin{frame}{Equilibrium System}

\footnotesize
\setlength{\abovedisplayskip}{1pt}
\setlength{\belowdisplayskip}{1pt}
\setlength{\abovedisplayshortskip}{0pt}
\setlength{\belowdisplayshortskip}{1pt}
\setlength{\parskip}{0pt}

\textbf{Phillips curve}:

\[
\bm\pi_t=\rho(I-\mathcal V)\mathbb E_t\bm\pi_{t+1}+\mathcal B(\gamma+\varphi)\tilde y_t-\mathcal V\bm\chi_t+\Gamma\Delta e_t
\]

\textbf{IS curve}:

\[
\tilde y_t=\mathbb E_t\tilde y_{t+1}-\frac{1}{\gamma}\big(i_t-\mathbb E_t\pi_t^C-r_t^{\mathrm{nat}}\big)-\frac{\eta(1-\omega)}{\gamma}\mathbb E_t\Delta\big(\hat s_t+\hat p_t^{F*}-\hat p_t^H\big)
\]

\textbf{Foreign sector}:

\[
i_t=i^*+\mathbb E_t\Delta e_{t+1}+\psi(\hat b_t^*)+RP_t,
\qquad
\hat b_t^*=\frac{1+i_{t-1}}{\Pi_t^C}\hat b_{t-1}^*+\frac{P_t^H\sum_iEX_{it}-P_t^{FH}IM_t}{P_t^C}
\]

\textbf{Policy rule}:

\[
\hat i_t=
\begin{cases}
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t & \text{Float}\\[2pt]
\text{residual (}\hat s_t=0\text{)} & \text{Peg}\\[2pt]
\phi_\pi\pi_t^{DC}+\phi_y\tilde y_t+\phi_s\hat s_t & \text{Managed}
\end{cases}
\]

{\scriptsize where: $r_t^{\mathrm{nat}}$ natural real rate, $\hat p_t^H$ log home price idx., $\pi_t^{DC}$ DC index (target); $\mathcal B,\Gamma,\mathcal V,\psi(\cdot),RP_t$ as before.}

\textbf{Welfare loss}: $\mathcal{W}=\tfrac12\big[(\gamma+\varphi)\mathrm{Var}(\tilde y_t)+\sum_i\kappa_i\mathrm{Var}(\pi_{it})\big]$, $\kappa_i=\lambda_{D,i}\varepsilon\frac{1-\hat\delta_i}{\hat\delta_i}$

\end{frame}

\begin{frame}{Mechanism: How the Network Amplifies Shocks}
%==============================================================

{\footnotesize Output-gap variance, \textbf{Network ON} vs.\ \textbf{Network OFF} (real Chile calibration) --- full IRFs: \hyperlink{netirfs}{Appendix}:}

\vspace{4pt}
\begin{center}
{\scriptsize
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{@{}lcc@{}}
\toprule
& \textbf{Import-price shock} & \textbf{Risk-premium/UIP shock} \\
\textbf{Regime} & Var($\tilde y_t$) ratio, ON/OFF & Var($\tilde y_t$) ratio, ON/OFF \\
\midrule
Float   & $5.4\times$  & $8.5\times$ \\
Managed & $6.0\times$  & $1.1\times$ (\emph{essentially none}) \\
Peg     & $11.1\times$ & $4.8\times$ \\
\bottomrule
\end{tabular}}
\end{center}

\vspace{5pt}
{\footnotesize
\begin{itemize}\setlength{\itemsep}{3pt}\setlength{\topsep}{0pt}
\item \textbf{Amplification is regime-\emph{and}-shock-specific, not a generic network multiplier.} Import-price shocks hit \textbf{Peg} hardest (no FX buffer to absorb pass-through); risk-premium shocks hit \textbf{Float} hardest ($\hat s_t$ free to move transmits the shock straight through the network) --- \textbf{Managed} blocks amplification of the risk-premium shock almost entirely, but not the import-price shock
\item \textbf{DC inflation variance is unchanged or slightly \emph{lower}} with the network on, for \emph{both} shocks in every regime --- this isn't new cost-push, it's a redirection of the existing shock into output
\item \textbf{Takeaway:} which shock dominates a regime's loss (Welfare Ranking, above) depends on which margin that regime leaves free to adjust
\end{itemize}}

\end{frame}

\begin{frame}{Openness \& Import Concentration}
%==============================================================
\vspace{-7pt}
{\scriptsize Reparametrize $\bm\omega_F$ only, on the REAL Chile calibration ($\alpha_i=1-\Omega^H_i-\omega_{F,i}$ absorbs it, holding $\Omega^H$ fixed):
\[
\omega_{F,i}(\zeta)=\zeta\,\omega_{F,i}^{\text{base}} \qquad\quad \omega_{F,i}(\theta_M)=\bar\omega_F+\theta_M\big(\omega_{F,i}^{\text{base}}-\bar\omega_F\big)
\]
$\zeta=\theta_M=1$: baseline. $\zeta$: uniform rescaling. $\theta_M$: mean-preserving redistribution ($\theta_M{=}0$: uniform exposure).}

\vspace{1pt}
\begin{center}
\includegraphics[width=0.85\textwidth]{figs/openness_concentration.pdf}
\end{center}

\vspace{-4pt}
{\scriptsize \textbf{Managed $<$ Float $<$ Peg robust throughout} (aggregate-export specification); gap moves in \textbf{opposite directions}:
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\topsep}{0pt}
\item $\zeta$ (openness): Peg $\downarrow$ ($153.7\to65.8$), Float $\uparrow$ ($23.1\to36.0$) --- \textbf{converge}
\item $\theta_M$ (concentration): Peg $\uparrow$ faster ($94.0\to107.3$) than Float ($25.2\to25.8$) --- \textbf{gap diverges}, via Peg rising, not Float falling
\item \textbf{Mechanism:} $\alpha_i+\Omega^H_i+\omega_{F,i}=1$ links labor- and import-cost channels; Peg's risk-premium/output-gap channel is far more sensitive to reallocation toward Services than Float's price-dispersion channel
\end{itemize}}

\end{frame}

$\zeta=\theta_X=1$: baseline. $\zeta$: uniform export-openness scaling. $\theta_X$: mean-preserving redistribution ($\theta_X{=}0$: uniform export exposure).}

\vspace{-4pt}
{\scriptsize \textbf{Managed $<$ Float $<$ Peg robust throughout.} Unlike import $\zeta$ (opposite directions --- $\omega_{F,i}$ trades off against $\alpha_i$), export $\zeta$ moves \textbf{all three regimes together}:
\begin{itemize}\setlength{\itemsep}{0pt}\setlength{\topsep}{0pt}
\item $\zeta$ (openness): \textbf{all regimes improve} as trade opens --- Peg most ($98.7\to84.8$), since $\kappa_{EX,i}$ only enters export demand, not the cost side
\item $\theta_X$ (concentration): \textbf{opposite directions, as with import concentration} --- Peg $\downarrow$ ($94.1\to89.6$), Float/Managed $\uparrow$ slightly, as exports concentrate toward Resource (the flexible-price sector)

\end{itemize}}

\section{Conclusion}
%==============================================================
\begin{frame}{Conclusion}
%==============================================================

{\footnotesize
\begin{itemize}\setlength{\itemsep}{6pt}
\item \textbf{Networks change how the exchange rate operates:} FX pass-through and TFP cost-push travel through the \emph{same} rigidity-adjusted network operator --- once $S_t$ is a second cost-push channel, the closed-economy divine-coincidence index generically fails
\item \textbf{Networks matter for severity and incidence, not the ranking:} switching the domestic network on substantially raises welfare losses in every regime and reallocates which sector bears them --- but \textbf{Managed $<$ Float $<$ Peg either way}
\item \textbf{Peg is dominated by a different mechanism than the textbook story:} not terms-of-trade, but risk-premium/UIP --- zero monetary autonomy forces the shock straight into the output gap
\item \textbf{Managed float wins} because it is the only regime that dampens the FX pass-through channel without fully reimposing peg's loss of autonomy
\item \textbf{Robust} across three real-data calibrations (Chile, Korea, Czechia) and a genuine second-order solution
\end{itemize}}

\end{frame}
