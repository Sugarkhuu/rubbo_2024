\section{Motivation}
%==============================================================
\begin{frame}{Motivation: Networks Meet the Exchange Rate}
%==============================================================

{\footnotesize \textbf{Two literatures that rarely talk to each other:}}

\vspace{4pt}
{\footnotesize
\begin{enumerate}\setlength{\itemsep}{4pt}
\item \textbf{Production-network Phillips curves} (Rubbo 2024; La'O \& Tahbaz-Salehi 2022): with input-output linkages, sector-level price stickiness generates a genuine inflation--output tradeoff for \emph{almost every} price index --- except one. A unique ``divine coincidence'' (DC) index, weighting sectors by Domar-scaled stickiness, restores the tradeoff-free policy prescription. All of this is derived in a \textbf{closed economy}.

\item \textbf{Small open economy NK models} (Gal\'i \& Monacelli 2005 and successors): the exchange rate is a cost-push shifter and a policy instrument, but the production side is a single aggregate sector --- no network structure, no heterogeneity in who is exposed to FX shocks.
\end{enumerate}}

\vspace{6pt}
{\footnotesize \textbf{This paper puts them together.} Once imported inputs enter a multi-sector network, the exchange rate becomes a \emph{second} cost-push channel that hits sectors asymmetrically depending on their import centrality and their position in the domestic supply chain. Does the DC index --- or any analog --- survive, and which FX regime is actually optimal once network structure is taken seriously?}

\end{frame}

%==============================================================
\begin{frame}{Why This Matters: The Policy Question}
%==============================================================

{\footnotesize \textbf{Motivating case: a small oil-exporting economy.}
Resource sector is upstream, import-intensive (machinery) and export-heavy;
Manufacturing sits in the middle; Services is downstream, domestically
oriented, and dominates final consumption.}

\vspace{6pt}
{\footnotesize \textbf{The question a central bank actually faces:}}
\begin{itemize}\setlength{\itemsep}{3pt}
\item \footnotesize Float and target inflation $\Rightarrow$ exchange rate absorbs FX/commodity-price shocks, but at the cost of nominal volatility and possibly balance-sheet risk.
\item \footnotesize Peg $\Rightarrow$ eliminates FX risk, but forces \emph{all} adjustment onto domestic prices and output --- amplified by whichever sectors are most import-exposed and most central in the network.
\item \footnotesize Manage the float $\Rightarrow$ partially, but at what welfare cost relative to the theoretical optimum?
\end{itemize}

\vspace{6pt}
{\footnotesize \textbf{Rubbo's closed-economy answer} (target the DC index) says nothing about \emph{which} exchange-rate regime to combine it with, or how network position shapes the answer. That is the gap this paper fills.}

\end{frame}

%==============================================================
\begin{frame}{Headline Result: Managed Float Wins the Regime Comparison}
%==============================================================

{\footnotesize \textbf{Jump to the answer first, mechanism after:} solving the full nonlinear open-economy network model for all three FX regimes gives a clear --- and non-obvious --- welfare ranking.}

\vspace{4pt}
\begin{columns}[t]
\column{0.46\textwidth}
\begin{center}
{\tiny \textbf{Total welfare loss ($\times10^4$, first-order moments)}}\\[2pt]
\begin{tikzpicture}
\begin{axis}[slide fig, xbar, width=6.3cm, height=4.0cm, bar width=13pt,
  symbolic y coords={Peg,Managed,Float}, ytick=data,
  xmin=0, xmax=12.5, xlabel={\tiny welfare loss ($\times10^4$)}, nodes near coords,
  nodes near coords style={font=\tiny}]
\addplot[fill=customgreen!45!white, draw=customgreen!70!black] coordinates {(10.48,Float) (4.19,Managed) (8.67,Peg)};
\end{axis}
\end{tikzpicture}
\end{center}
\column{0.50\textwidth}
{\footnotesize \textbf{Managed float dominates both corner regimes} --- less than half the loss of free float, and lower than a hard peg too. This is the model's own solution to the FX-regime choice, not an assumption.}

\vspace{6pt}
{\footnotesize \textbf{Why this is surprising:} float directly targets the divine-coincidence index and should, on the textbook intuition, be (weakly) best. It is not --- floating turns the exchange rate into a slow, persistent state variable that keeps feeding cost-push shocks back into the network (mechanism in the Welfare section).}
\end{columns}

\vspace{6pt}
{\footnotesize \textbf{Roadmap:} Literature $\to$ This paper $\to$ Model $\to$ Theory (why DC targeting could fail to survive) $\to$ Calibration $\to$ Shocks/IRFs $\to$ network amplification $\to$ welfare cost $\to$ shock decomposition.}

\end{frame}

\section{Literature}
%==============================================================
\begin{frame}{Literature}
%==============================================================

{\footnotesize
\begin{itemize}\setlength{\itemsep}{5pt}
\item \textbf{La'O \& Tahbaz-Salehi (2022, \emph{Econometrica})} --- optimal monetary policy in production networks; efficient allocations require targeting a Domar-weighted index, not simple CPI. Closed economy.
\item \textbf{Rubbo (2024, \emph{Econometrica})} --- the paper we extend. Sectoral Calvo pricing $+$ IO network $\Rightarrow$ unique DC index with \emph{no} cost-push residual; welfare-optimal Taylor rule targets it.
\item \textbf{Auer, Levchenko \& Sauré (2019, \emph{ReStud})} --- international inflation spillovers propagate through global IO linkages; empirical evidence that network position matters for cross-border pass-through.
\item \textbf{Pasten, Schoenle \& Weber (2020, \emph{JPE})} --- heterogeneous price rigidity across sectors reshapes monetary transmission; motivates letting $\hat\delta_i$ differ by sector here too.
\item \textbf{Fanelli \& Straub (2021, \emph{JPE})} --- theory of sterilized FX intervention as a second instrument alongside the policy rate.
\item \textbf{Gopinath \& Itskhoki (2022)} --- dominant currency pricing; \textbf{Amiti, Itskhoki \& Konings (2019, \emph{QJE})} --- variable markups shape exchange-rate pass-through at the firm level.
\end{itemize}}

\vspace{4pt}
{\footnotesize \textbf{Gap:} no existing paper combines a multi-sector Calvo network \emph{and} a genuine open-economy block (UIP, imported inputs, FX regimes) to ask which FX regime is optimal, and whether the DC-targeting result survives.}

\end{frame}

%==============================================================
\section{This Paper}
%==============================================================
\begin{frame}{This Paper: Contribution and Approach}
%==============================================================

{\footnotesize \textbf{What we do:}}
\begin{itemize}\setlength{\itemsep}{4pt}
\item \footnotesize Extend Rubbo's network Phillips curve to a small open economy: imported intermediate inputs (with an import-cost-share matrix $\Omega^F$), imported final consumption, UIP with a debt-elastic premium.
\item \footnotesize Ask which FX regime (float / peg / managed float) is welfare-optimal in a multi-sector network, and whether the DC index survives when the exchange rate is a second cost-push channel.
\item \footnotesize Calibrate to a 3-sector oil-exporting economy (Resource $\to$ Manufacturing $\to$ Services).
\end{itemize}

\vspace{8pt}
{\footnotesize \textbf{Methodological choice:} the model is built and solved as the \emph{original nonlinear} system --- Cobb-Douglas cost minimization, the exact recursive Calvo pricing problem, CRRA/Frisch household FOCs, CES consumption aggregation --- rather than substituting in a hand-linearized Phillips curve. The nonlinear steady state is computed numerically and the system perturbed around it, so the log-linear coefficients (e.g.\ $\hat\delta_i$) come out of the solution rather than being assumed.}

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Households --- Preferences and Budget}
%==============================================================

{\footnotesize \textbf{Preferences} (CRRA consumption, Frisch labor supply; King, Plosser \& Rebelo 1988):}
\vspace{-4pt}
{\scriptsize\[
\max\ \mathbb{E}_0\sum_{t=0}^\infty \beta^t\left[\frac{C_t^{1-\gamma}}{1-\gamma} - \frac{L_t^{1+\varphi}}{1+\varphi}\right]
\]}

\vspace{1pt}
{\footnotesize \textbf{Sequential budget constraint} (representative household; $\Pi_t$ = firm profits rebated lump-sum; $B^*_t$ = real holdings of the internationally traded bond, valued at $S_t$). The household chooses $\{C_t,L_t,B^*_t\}_{t\ge0}$ subject to this each period:}
\vspace{-4pt}
{\scriptsize\[
P^C_tC_t + S_tB^*_t = W_tL_t + \Pi_t + (1{+}i^*)\big[1{-}\psi(\hat b^*_{t-1}{-}\bar b^*)\big]S_tB^*_{t-1}
\]}

\vspace{1pt}
{\footnotesize \textbf{No domestic bond needed:} representative household, net-zero supply $\Rightarrow$ redundant asset (Wallace irrelevance); the internationally traded bond alone prices consumption smoothing.}

\vspace{1pt}
{\footnotesize \textbf{Consumption aggregation:} CES nest between a domestic bundle and an imported bundle, home-bias $\omega$, elasticity $\eta$ (Armington 1969; Gal\'i \& Monacelli 2005); the domestic bundle is Cobb-Douglas across sectors with shares $\beta^H_i$:}
\vspace{-4pt}
{\scriptsize\[
C_t=\Big[\omega^{1/\eta}(C^H_t)^{\frac{\eta-1}{\eta}}+(1-\omega)^{1/\eta}(C^F_t)^{\frac{\eta-1}{\eta}}\Big]^{\frac{\eta}{\eta-1}},\qquad
C^H_t = \prod_i \Big(\tfrac{C_{it}}{\beta^H_i}\Big)^{\beta^H_i}
\]}

\vspace{1pt}
{\footnotesize \textbf{Dual price indices} (cost of one unit of $C_t^H$ and $C_t$ respectively; $P^{FH}_t$ defined on the Open-Economy slide):}
\vspace{-4pt}
{\scriptsize\[
P^H_t=\prod_iP_{it}^{\beta^H_i},\qquad
P^C_t=\big[\omega(P^H_t)^{1-\eta}+(1{-}\omega)(P^{FH}_t)^{1-\eta}\big]^{\frac{1}{1-\eta}}
\]}

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Households --- Optimality Conditions}
%==============================================================

{\footnotesize \textbf{FOCs} (from the Lagrangian on the budget constraint, choice variables $C_t,L_t,B^*_t$; exact nonlinear conditions):}

\vspace{4pt}
{\footnotesize \textbf{Euler equation} ($\partial/\partial C_t$, combined with $\partial/\partial C_{t+1}$):}
\vspace{-4pt}
\[
C_t^{-\gamma} = \beta(1+i_t)\,\mathbb{E}_t\big[C_{t+1}^{-\gamma}/\Pi^C_{t+1}\big]
\]

\vspace{3pt}
{\footnotesize \textbf{Labor supply} ($\partial/\partial L_t$):}
\vspace{-4pt}
\[
W_t/P^C_t = C_t^{\gamma} L_t^{\varphi}
\]

\vspace{3pt}
{\footnotesize \textbf{Foreign-bond FOC} ($\partial/\partial B^*_t$):}
\vspace{-4pt}
\[
C_t^{-\gamma} = \beta(1{+}i^*)\big[1{-}\psi(\hat b^*_t{-}\bar b^*)\big]\,\mathbb{E}_t\Big[C_{t+1}^{-\gamma}\,\tfrac{S_{t+1}}{S_t\Pi^C_{t+1}}\Big]
\]

\vspace{4pt}
{\footnotesize \textbf{Where UIP comes from:} eliminating the common term $\mathbb{E}_t\big[C_{t+1}^{-\gamma}/\Pi^C_{t+1}\big]$ between the Euler equation and the foreign-bond FOC above gives exactly the uncovered interest parity condition used throughout the open-economy block --- it is \emph{derived} from household optimization, not assumed:}
\vspace{-4pt}
\[
(1+i_t) = (1+i^*)\big[1-\psi(\hat b^*_t-\bar b^*)\big]\,\mathbb{E}_t\big[S_{t+1}/S_t\big]
\]

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Firms --- Technology and Cost Minimization}
%==============================================================

{\footnotesize \textbf{Production} (Cobb-Douglas, sector $i$; network structure as in La'O \& Tahbaz-Salehi 2022, Rubbo 2024):}
\vspace{-4pt}
\[
Y_{it}=A_{it}\,L_{it}^{\alpha_i}\prod_j X_{ijt}^{\Omega^H_{ij}}\,M_{it}^{\Omega^F_i}
\]

\vspace{2pt}
{\footnotesize \textbf{Cost minimization:} choose $\{L_{it},X_{ijt},M_{it}\}$ to minimize total cost $W_tL_{it}+\sum_jP_{jt}X_{ijt}+S_tP^{F*}_tM_{it}$ subject to the production constraint above. The FOCs equate each input's expenditure to its Cobb-Douglas exponent times total cost $MC_{it}Y_{it}$:}
\vspace{-4pt}
\[
W_tL_{it}=\alpha_i MC_{it}Y_{it},\qquad P_{jt}X_{ijt}=\Omega^H_{ij}MC_{it}Y_{it},\qquad S_tP^{F*}_tM_{it}=\Omega^F_i MC_{it}Y_{it}
\]

\vspace{2pt}
{\footnotesize Summing the three input demands back into the cost function gives nominal marginal cost, multiplicatively separable in input prices:}
\vspace{-4pt}
\[
MC_{it}=\frac{W_t^{\alpha_i}\prod_j P_{jt}^{\Omega^H_{ij}}\,(S_tP^{F*}_t)^{\Omega^F_i}}{A_{it}}
\]

\vspace{2pt}
{\footnotesize This is the object that both the exchange rate (via $M_{it}$, Open-Economy slide) and TFP shocks $A_{it}$ shift --- the two cost-push channels the rest of the paper is about.}

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Firms --- Calvo Price Setting}
%==============================================================

{\footnotesize \textbf{Calvo pricing} (Calvo 1983): with probability $\delta_i$ (primitive; \emph{not} $\hat\delta_i$, derived below) the firm resets its price each period; with probability $1{-}\delta_i$ it keeps last period's price. A resetting firm chooses $P_{it}^\#$ to maximize expected discounted profits, weighting each future state by the probability $(1-\delta_i)^s$ that the price is \emph{still} not reset $s$ periods later:}
\vspace{-4pt}
\[
\max_{P_{it}^\#}\ \mathbb{E}_t\sum_{s=0}^\infty\rho^s(1{-}\delta_i)^s\big[P_{it}^\#{-}MC_{it+s}\big]\Big(\tfrac{P_{it}^\#}{P_{it+s}}\Big)^{-\varepsilon}Y_{it+s}
\]

\vspace{2pt}
{\footnotesize where $\big(P_{it}^\#/P_{it+s}\big)^{-\varepsilon}Y_{it+s}$ is the CES demand curve the firm faces (elasticity $\varepsilon$, markup $\varepsilon/(\varepsilon{-}1)$). FOC (forward-looking markup pricing):}
\vspace{-4pt}
\[
P_{it}^{\#}=\frac{\varepsilon}{\varepsilon-1}\cdot\frac{\mathbb{E}_t\sum_s \rho^s(1-\delta_i)^s Y_{it+s}\,\widetilde{MC}_{it+s}}{\mathbb{E}_t\sum_s \rho^s(1-\delta_i)^s Y_{it+s}/P_{it+s}}
\]

\vspace{2pt}
{\footnotesize \textbf{Reading the FOC:} the reset price is a markup over a \emph{weighted average of current and future marginal costs}, with weights given by how likely the price is to still be in effect. The stickier the sector ($\delta_i$ small), the further forward the firm must look --- this forward-lookingness is exactly what the recursive form on the next slide computes period by period.}

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Firms --- Calvo Recursion, as Solved}
%==============================================================

{\footnotesize \textbf{As coded} (recursive form of the FOC on the previous slide, real MC $\widetilde{MC}=MC/P$): $X1_{it}$ and $X2_{it}$ are the numerator and denominator of the reset-price formula, written recursively so Dynare can solve them jointly with the rest of the nonlinear system:}
\vspace{-4pt}
\[
X1_{it} = \widetilde{MC}_{it}Y_{it} + (1{-}\delta_i)\beta\Big(\tfrac{C_{t+1}}{C_t}\Big)^{-\gamma}\Pi_{i,t+1}^{\varepsilon}X1_{i,t+1}
\]
\vspace{-10pt}
\[
X2_{it} = Y_{it} + (1{-}\delta_i)\beta\Big(\tfrac{C_{t+1}}{C_t}\Big)^{-\gamma}\Pi_{i,t+1}^{\varepsilon-1}X2_{i,t+1},
\qquad
P_{it}^\#/P_{it} = \tfrac{\varepsilon}{\varepsilon-1}\,X1_{it}/X2_{it}
\]

\vspace{4pt}
{\footnotesize \textbf{Price index law of motion} (CES aggregation of the $\delta_i$ fraction that resets to $P_{it}^\#$ and the $1{-}\delta_i$ fraction that keeps last period's price):}
\vspace{-4pt}
\[
1 = (1{-}\delta_i)\Pi_{it}^{\varepsilon-1} + \delta_i \big(P_{it}^\#/P_{it}\big)^{1-\varepsilon}
\]

\vspace{4pt}
{\footnotesize \textbf{Why this matters:} log-linearizing this exact recursion around the steady state is what \emph{produces} Rubbo's adjusted slope $\hat\delta_i=\tfrac{\delta_i(1-\beta(1-\delta_i))}{1-\beta\delta_i(1-\delta_i)}$, which appears throughout the Theory section --- it is a derived object, not a primitive of the model.}

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Open Economy --- Prices and Interest Parity}
%==============================================================

{\footnotesize \textbf{Law of one price for imports} ($S_t$: nominal ER, home ccy.\ per unit foreign ccy.; $P^{F*}_t$: foreign-currency import price; Gal\'i \& Monacelli 2005):}
\vspace{-4pt}
{\scriptsize\[
P^{FH}_t = S_t P^{F*}_t
\]}
{\footnotesize This is the last piece needed to close the price indices from the Households slide: $P^H_t=\prod_iP_{it}^{\beta^H_i}$ is the price of the domestic bundle, and $P^C_t=\big[\omega(P^H_t)^{1-\eta}+(1{-}\omega)(P^{FH}_t)^{1-\eta}\big]^{1/(1-\eta)}$ is the CPI.}

\vspace{6pt}
{\footnotesize \textbf{Uncovered interest parity, with a debt-elastic risk premium} (functional form: Schmitt-Groh\'e \& Uribe 2003, for stationarity of $\hat b^*_t$). Derived, not assumed: eliminate $\mathbb{E}_tC_{t+1}^{-\gamma}/\Pi^C_{t+1}$ between the household's domestic Euler equation and its foreign-bond FOC (Households slide) to get}
\vspace{-4pt}
{\scriptsize\[
(1+i_t) = (1+i^*)\big[1-\psi(\hat b^*_t-\bar b^*)\big]\,\mathbb{E}_t\big[S_{t+1}/S_t\big]
\]}
{\footnotesize Under a float, $i_t$ follows the Taylor rule and $S_t$ adjusts to satisfy this equation; under a peg, $S_t=\bar S$ is fixed and $i_t$ is the residual.}

\vspace{6pt}
{\footnotesize \textbf{Key mechanism:} imported inputs enter every sector's marginal cost through $\Omega^F_i$ (Firms slide, via $M_{it}$) --- the exchange rate now moves marginal cost \emph{exactly like a productivity shock}, scaled by each sector's import exposure and propagated through $\Omega^H$.}

\end{frame}

%==============================================================
\begin{frame}{Model Setup: Open Economy --- External Accounts and Market Clearing}
%==============================================================

{\footnotesize \textbf{Net foreign assets} (aggregating household budget constraints across the economy collapses to the standard current-account law of motion):}
\vspace{-4pt}
{\scriptsize\[
\hat b^*_t = \frac{1+i_{t-1}}{\Pi^C_t}\hat b^*_{t-1} + \frac{P^H_t EX_t - P^{FH}_t IM_t}{P^C_t}
\]}

\vspace{4pt}
{\footnotesize \textbf{Trade flows:} $EX_t$ is reduced-form foreign demand for the home bundle (elasticity $\theta^*$, shifter $D^*_t$); $IM_t$ sums final-consumption imports and the imported input embedded in every sector's cost:}
\vspace{-4pt}
{\scriptsize\[
EX_t = \kappa_{EX}\,D^*_t\Big(\frac{P^H_t}{P^{FH}_t}\Big)^{-\theta^*},\qquad
IM_t = C^F_t + \sum_i \Omega^F_i\frac{MC_{it}Y_{it}}{P^{FH}_t}
\]}

\vspace{4pt}
{\footnotesize \textbf{Sectoral goods-market clearing:} sector $i$'s output is absorbed by domestic consumption and exports of $i$'s share of the bundle, plus intermediate demand from every downstream sector $j$ that uses $i$ as an input:}
\vspace{-4pt}
{\scriptsize\[
Y_{it} = \beta^H_i\,\frac{P^H_t}{P_{it}}\big(C^H_t+EX_t\big) + \sum_j \Omega^H_{ji}\,\frac{MC_{jt}Y_{jt}}{P_{it}}
\]}
{\footnotesize (In the 3-sector calibration: Resource feeds Manufacturing, Manufacturing feeds Services, and each sector also sells directly to $C^H_t$ and $EX_t$.)}

\end{frame}

\section{Theory}

\begin{frame}{Rubbo (2024): What Carries Over}
%==============================================================

\begin{columns}[t]
\column{0.56\textwidth}

{\footnotesize \textbf{Closed-economy system} ($N$ sectors, Calvo $\hat\delta_i$, IO network $\Omega$):}
\vspace{-3pt}
{\scriptsize\begin{align*}
\text{IS:} &\quad \ty_t = \mathbb{E}\ty_{t+1} - \tfrac{1}{\gamma}(i_t - \mathbb{E}\pi^C_{t+1} - r^{nat}_t) \\[1pt]
\text{MC:} &\quad \bm{mc}_t = \bm{\alpha}w_t + \Omega\bm{p}_t - \log\mathbf{A}_t \\[1pt]
\text{NKPC:} &\quad \bpi_t = \rho(I{-}\mathcal{V})\mathbb{E}\bpi_{t+1} + \mathcal{B}(\gamma{+}\varphi)\ty_t - \mathcal{V}\bm{\chi}_t \\[1pt]
\text{Policy:} &\quad i_t = r^{nat}_t + \phi_\pi\pi^C_t + \phi_y\ty_t
\end{align*}}

\vspace{3pt}
{\footnotesize \textbf{Lemma 1:} natural output $=$ Domar-weighted TFP, $\;y^{nat}_t=\sum_i\lambda_i a_{it}$.

\textbf{Proposition 1 (DC index):} unique index with \emph{no} cost-push:}
\vspace{-4pt}
\[\footnotesize \pi^{DC}_t = \rho\,\mathbb{E}\pi^{DC}_{t+1} + \tfrac{\gamma+\varphi}{\Lambda^{DC}}\ty_t,\quad w^{DC}_i \propto \lambda_i\tfrac{1-\hat\delta_i}{\hat\delta_i}\]

\column{0.40\textwidth}

{\footnotesize \textbf{Why DC is unique:} $\mathcal{V}=\Delta(I-\Omega\Delta)^{-1}$ has rank $N{-}1$ on the space of cost-push shocks $\Rightarrow$ a one-dimensional null space $\Rightarrow$ exactly one index $\bm\phi$ with $\bm\phi^\top\mathcal{V}=\bm 0^\top$.}

\vspace{6pt}
{\footnotesize \textbf{One shock, one channel:} in the closed economy, only TFP shocks $\bm\chi_t$ enter the cost-push term. That is exactly why a single index can null it out.}

\vspace{6pt}
{\footnotesize \textbf{The open question we ask:} with a \emph{second} cost-push channel (the exchange rate), can the same trick still null out both simultaneously?}

\end{columns}
\end{frame}

%==============================================================
\begin{frame}{Why the Divine Coincidence Matters}
%==============================================================

{\scriptsize \textbf{Where the name comes from} (Blanchard \& Gal\'i 2007, \emph{JEEA}): in the canonical one-sector NK model, if the \emph{only} distortion is nominal price stickiness, then stabilizing inflation, closing the output gap, and maximizing welfare are the \emph{same} objective --- one instrument, one target, no tradeoff. It is called ``divine'' because this is fragile: any second distortion (real wage rigidity, markup or cost-push shocks) breaks it and forces a genuine tradeoff.}

\vspace{2pt}
{\scriptsize \textbf{Why sectoral heterogeneity threatens it:} with $N$ sectors resetting prices at different rates $\delta_i$, an arbitrary index --- plain CPI, say --- behaves like a model with an extra distortion: stabilizing it no longer stabilizes the output gap. Rubbo's Proposition 1 shows the coincidence is not lost, only \emph{hidden}: exactly one Domar- and stickiness-weighted index restores it (previous slide).}

\vspace{2pt}
{\scriptsize \textbf{Why it matters in practice:}}
\begin{itemize}\setlength{\itemsep}{0pt}
\item \scriptsize It gives a central bank a single, well-defined, model-consistent target whose stabilization \emph{alone} delivers the fully efficient (flexible-price) allocation --- no discretionary weighing of inflation against output.
\item \scriptsize It shows plain CPI targeting is \emph{not} optimal once sectors differ in stickiness: the wedge between CPI inflation and $\pi^{DC}_t$ is exactly the welfare-relevant cost-push term a CPI-targeting central bank would be missing.
\item \scriptsize It is not merely a price index: by Proposition 3, $\pi^{DC}_t\equiv0$ for all $t$ exactly minimizes the model's quadratic welfare loss --- ``divine coincidence'' and ``welfare-optimal policy'' are the same statement.
\end{itemize}

\vspace{2pt}
{\scriptsize \textbf{Why it recurs on almost every slide that follows:} the paper is organized around one question --- does this single tradeoff-free target survive once the exchange rate becomes a \emph{second} source of cost-push shocks. Every quantitative result below is, implicitly, a test of how ``divine'' the coincidence remains once the economy is open.}

\end{frame}

%==============================================================
\begin{frame}{Open-Economy Divine Coincidence: Does It Survive?}
%==============================================================

{\footnotesize \textbf{Closed economy (Rubbo):} one cost-push channel (TFP) $\Rightarrow$ DC index exists uniquely because a rank-deficient matrix $\mathcal{V}$ has a one-dimensional null space, and one index can sit in it.}

\vspace{6pt}
{\footnotesize \textbf{Open economy: two cost-push channels.} TFP \emph{and} the exchange rate now shift sectoral marginal cost. A single weight vector would need to null out \emph{both} simultaneously:}
\vspace{-4pt}
\[\footnotesize \underbrace{\bm\phi^\top\mathcal{V}=\bm 0^\top}_{\text{TFP channel}} \quad\text{and}\quad \underbrace{\bm\phi^\top\Gamma = 0}_{\text{FX channel}}\]
{\footnotesize Two conditions, one free vector $\Rightarrow$ generically no solution: \textbf{DC breaks down under a float.}}

\vspace{6pt}
{\footnotesize \textbf{But:} import nodes can be modeled as fully flexible upstream sectors ($\hat\delta_M=1$) in an augmented network. Fully flexible nodes get \emph{zero} DC weight ($w_M=0$ since $(1-\hat\delta_M)/\hat\delta_M=0$), so Rubbo's Proposition 1 still applies to the \emph{augmented} network --- the DC index (as coded here) remains well-defined and insulates against domestic TFP shocks by construction. Whether it \emph{also} nulls the FX channel in the calibrated network is the quantitative question tackled below.}

\end{frame}

\section{Calibration}
%==============================================================
\begin{frame}{Solution Method: Nonlinear, Not Pre-Linearized}
%==============================================================

{\footnotesize \textbf{Why solve the nonlinear model directly, rather than hand-linearize first:}}
\begin{itemize}\setlength{\itemsep}{4pt}
\item \footnotesize The Calvo slope $\hat\delta_i$ and the DC weights are \emph{outputs} of log-linearizing the true recursion --- coding them as inputs risks baking in the answer.
\item \footnotesize A genuine nonlinear model lets us go to order-2 perturbation, which is required for valid welfare comparisons across policy regimes (Kim-Kim / Woodford: first-order-accurate welfare rankings are not valid --- variance terms enter the welfare function only at second order).
\end{itemize}

\vspace{5pt}
{\footnotesize \textbf{Steady state:} solved semi-analytically, not guessed. At zero inflation, price = markup $\times$ marginal cost exactly, which collapses the 46-equation system to one scalar root-find in the real wage $W$ (prices, exports, consumption all closed-form given $W$), giving $W^*=1.3189$.}
\begin{itemize}\setlength{\itemsep}{2pt}
\item \footnotesize \textbf{Lemma 1 (exact):} $\tilde y_t=\sum_i\lambda_{D,i}\big[\log(Y_{it}/\bar Y_i)-\log A_{it}\big]$ --- no shadow flex-price block needed.
\item \footnotesize \textbf{Prop.\ 1 weights (theory-motivated):} DC weights from $\lambda_{D,i}$, $\hat\delta_i$, applied to realized inflation in the true nonlinear simulation.
\end{itemize}

\vspace{4pt}
{\footnotesize \textbf{Solved at first order} around the exact steady state, for all three regimes. Second-order perturbation is not used: an inflation/output-gap-only Taylor rule leaves the price \emph{level} and $S_t$ as a unit root --- expected, since nothing anchors the level --- so only the first-order (stationary) solution is used, and only stationary objects enter welfare.}

\end{frame}

\begin{frame}{Which Sector Prefers Which Regime? Two Equations}
%==============================================================

\vspace{-6pt}

\vspace{-6pt}

\section{Generalization}
%==============================================================
\begin{frame}{From ``An Oil Exporter'' to ``Which Structural Features Matter''}
%==============================================================

{\footnotesize \textbf{The calibration above is one economy.} The Theory section's result --- the DC index breaks down under a float because $\bm\phi^\top\Gamma=0$ and $\bm\phi^\top\mathcal{V}=\bm 0^\top$ generically cannot hold simultaneously --- is a statement about \emph{any} network with $\Gamma\not\propto\mathcal{B}$ (FX exposure not proportional to labor intensity). This section asks which structural features govern \emph{how costly} that breakdown is, using three re-solves of the full nonlinear model (not new shocks -- the same three regimes, the same welfare metric):}

\vspace{6pt}
\begin{enumerate}\setlength{\itemsep}{5pt}
\item \footnotesize \textbf{Policy-space sensitivity ($\phi_s$):} for a fixed economy, how much FX stabilization is optimal? (already-calibrated managed-float rule, re-solved on a grid.)
\item \footnotesize \textbf{Import-intensity level:} scale the whole import-cost-share vector $\Omega^F$ up/down, holding its \emph{relative} sectoral pattern fixed. Tests whether the peg-can-beat-float result is an artifact of this economy's degree of import dependence.
\item \footnotesize \textbf{Import-exposure heterogeneity:} hold \emph{total} import intensity fixed but vary how unevenly it is spread across sectors (Resource/Manuf./Services). Tests the theorem directly: less heterogeneity should move $\Gamma$ closer to proportional, and the float/peg welfare gap should shrink.
\end{enumerate}

\vspace{4pt}
{\footnotesize \textbf{Reading ahead:} if the managed-float result is fragile to (2)--(3), it is a fact about oil exporters; if it survives, it is a fact about production networks with heterogeneous FX exposure -- a much stronger claim.}

\end{frame}

%==============================================================
\section{Conclusion}
%==============================================================
\begin{frame}{Summary: Key Messages}
%==============================================================

{\footnotesize
\begin{enumerate}\setlength{\itemsep}{3pt}
\item \footnotesize \textbf{Free float is not the optimal regime, even though it directly targets the DC index.} Managed float has less than half the total welfare loss of pure float (4.19 vs.\ 10.48, $\times10^4$) and beats the hard peg too (8.67). This is the model's own solution, not an assumption.
\item \footnotesize \textbf{Why:} floating turns the exchange rate into a slow, persistent state variable (via UIP and the near-unit-root NFA process) that feeds continuously back into every sector's marginal cost through $\Omega^F_i$, turning even domestic TFP shocks into long, drawn-out inflation. \textbf{Persistence, not on-impact volatility, drives the welfare cost.}
\item \footnotesize \textbf{Every sector prefers managed float first}, but ranks the corner regimes oppositely: Resource (import-exposed, peripheral) ranks float over peg; Services (import-light, central) ranks peg over float.
\item \footnotesize \textbf{The optimal FX stabilization is small but strictly positive}: welfare is minimized around $\phi_s\approx0.2$--$0.3$, with more than half the gain from just $\phi_s=0.10$.
\item \footnotesize \textbf{The managed $<$ peg $<$ float ranking is robust} to import intensity ($\kappa\in[0.25,2.5]$) and exposure heterogeneity ($\theta\in[0,1.5]$) --- not an oil-exporter artifact. The heterogeneity gap moves opposite the naive $\Gamma\propto\mathcal{B}$ prediction since this design lets $\alpha_i$ absorb the $\Omega^F_i$ change; isolating that mechanism cleanly is future work.
\item \footnotesize These are first-order welfare comparisons (order-2 pruning fails due to the price-level/ER unit root) --- rankings are indicative of the mechanism, not certainty-equivalent-exact.
\end{enumerate}}

\end{frame}
